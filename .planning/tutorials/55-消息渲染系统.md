## 一、系统概述

消息渲染系统是 Claude Code CLI 的核心 UI 组件，负责将对话历史、工具调用、系统消息等渲染为终端界面。系统采用 React + Ink 架构，实现了高性能的虚拟滚动、流式渲染和增量更新。

### 关键特性

1. **虚拟滚动** - 支持数千条消息的高效渲染
2. **流式渲染** - 实时显示 AI 响应和工具执行过程
3. **消息折叠** - 智能合并重复消息，优化显示
4. **代码高亮** - 支持多种编程语言的语法高亮
5. **Markdown 渲染** - 完整的 GFM (GitHub Flavored Markdown) 支持

### 3.4 虚拟滚动实现

系统实现了两层虚拟滚动：

#### 1. 非虚拟化模式的安全上限

```typescript
// 限制渲染消息数量，防止内存溢出
const MAX_MESSAGES_WITHOUT_VIRTUALIZATION = 200
const MESSAGE_CAP_STEP = 50

// 基于UUID的锚点切片（比索引切片更稳定）
export function computeSliceStart(
  collapsed: ReadonlyArray<{ uuid: string }>,
  anchorRef: { current: SliceAnchor },
  cap = MAX_MESSAGES_WITHOUT_VIRTUALIZATION,
  step = MESSAGE_CAP_STEP
): number {
  const anchor = anchorRef.current
  const anchorIdx = anchor 
    ? collapsed.findIndex(m => m.uuid === anchor.uuid) 
    : -1
  
  // 找到锚点使用锚点，否则使用存储的索引
  let start = anchorIdx >= 0 
    ? anchorIdx 
    : anchor 
      ? Math.min(anchor.idx, Math.max(0, collapsed.length - cap))
      : 0
  
  // 超过上限时推进切片起点
  if (collapsed.length - start > cap + step) {
    start = collapsed.length - cap
  }
  
  return start
}
```

### 3.6 性能优化：React.memo 自定义比较

```typescript
export const Messages = React.memo(MessagesImpl, (prev, next) => {
  const keys = Object.keys(prev) as (keyof typeof prev)[]
  
  for (const key of keys) {
    // 跳过不影响渲染的回调函数
    if (key === 'onOpenRateLimitOptions' || 
        key === 'scrollRef' || 
        key === 'jumpRef') {
      continue
    }
    
    if (prev[key] !== next[key]) {
      // 特殊处理 streamingToolUses：只比较 contentBlock
      if (key === 'streamingToolUses') {
        const p = prev.streamingToolUses
        const n = next.streamingToolUses
        if (p.length === n.length && 
            p.every((item, i) => item.contentBlock === n[i]?.contentBlock)) {
          continue
        }
      }
      
      // 特殊处理 Set 比较
      if (key === 'inProgressToolUseIDs') {
        if (setsEqual(prev.inProgressToolUseIDs, next.inProgressToolUseIDs)) {
          continue
        }
      }
      
      return false
    }
  }
  
  return true
})
```

#### Token 缓存优化

```typescript
// 模块级 token 缓存，避免重复解析
const TOKEN_CACHE_MAX = 500
const tokenCache = new Map<string, Token[]>()

function cachedLexer(content: string): Token[] {
  // 快速路径：无 Markdown 语法的纯文本
  if (!hasMarkdownSyntax(content)) {
    return [{
      type: 'paragraph',
      raw: content,
      text: content,
      tokens: [{ type: 'text', raw: content, text: content }]
    }]
  }
  
  // 缓存查找
  const key = hashContent(content)
  const hit = tokenCache.get(key)
  if (hit) {
    // LRU：提升到最近使用
    tokenCache.delete(key)
    tokenCache.set(key, hit)
    return hit
  }
  
  // 解析并缓存
  const tokens = marked.lexer(content)
  if (tokenCache.size >= TOKEN_CACHE_MAX) {
    const first = tokenCache.keys().next().value
    if (first !== undefined) tokenCache.delete(first)
  }
  tokenCache.set(key, tokens)
  return tokens
}
```

```typescript
export function StreamingMarkdown({ children }: { children: string }) {
  'use no memo'  // 禁用 React Compiler 优化
  
  const stablePrefixRef = useRef('')
  const prevLengthRef = useRef(0)
  
  configureMarked()
  const stripped = stripPromptXMLTags(children)
  
  // 检测内容是否被重置（新消息）
  if (!stripped.startsWith(stablePrefixRef.current)) {
    stablePrefixRef.current = ''
    prevLengthRef.current = 0
  }
  
  // 找到最后一个完整块的边界
  let boundaryIndex = findLastBlockBoundary(stripped)
  const stablePart = stripped.slice(0, boundaryIndex)
  const streamingPart = stripped.slice(boundaryIndex)
  
  // 只有稳定部分推进时才更新
  if (stablePart.length > stablePrefixRef.current.length) {
    stablePrefixRef.current = stablePart
  }
  
  // 稳定部分使用缓存渲染，流式部分重新解析
  return (
    <Box flexDirection="column">
      {stablePart && <Markdown>{stablePart}</Markdown>}
      {streamingPart && <Markdown>{streamingPart}</Markdown>}
    </Box>
  )
}
```

### 4.2 消息标准化

`utils/messages.ts` 中的 `normalizeMessages` 函数负责将消息拆分为单内容块：

```typescript
export function normalizeMessages(messages: Message[]): NormalizedMessage[] {
  let isNewChain = false
  
  return messages.flatMap(message => {
    switch (message.type) {
      case 'assistant': {
        isNewChain = isNewChain || message.message.content.length > 1
        
        // 将多内容块拆分为多个单内容块消息
        return message.message.content.map((block, index) => {
          const uuid = isNewChain
            ? deriveUUID(message.uuid, index)  // 生成确定性 UUID
            : message.uuid
          
          return {
            type: 'assistant' as const,
            timestamp: message.timestamp,
            message: {
              ...message.message,
              content: [block],  // 单个内容块
            },
            uuid,
            // ... 其他字段
          }
        })
      }
      
      case 'user': {
        if (typeof message.message.content === 'string') {
          // 字符串内容转换为文本块
          return [{
            ...message,
            message: {
              ...message.message,
              content: [{ type: 'text', text: message.message.content }]
            }
          }]
        }
        
        // 数组内容同样拆分
        return message.message.content.map((block, index) => ({
          ...createUserMessage({ content: [block] }),
          uuid: isNewChain ? deriveUUID(message.uuid, index) : message.uuid,
        }))
      }
      
      default:
        return [message]
    }
  })
}
```

### 4.3 消息重排序

工具调用和结果消息需要正确配对和排序：

```typescript
export function reorderMessagesInUI(
  messages: (NormalizedUserMessage | NormalizedAssistantMessage | AttachmentMessage | SystemMessage)[]
): (NormalizedUserMessage | NormalizedAssistantMessage | AttachmentMessage | SystemMessage)[] {
  // 按工具调用 ID 分组
  const toolUseGroups = new Map<string, {
    toolUse: ToolUseRequestMessage | null
    preHooks: AttachmentMessage[]
    toolResult: NormalizedUserMessage | null
    postHooks: AttachmentMessage[]
  }>()
  
  // 第一遍：分组
  for (const message of messages) {
    if (isToolUseRequestMessage(message)) {
      const toolUseID = message.message.content[0]?.id
      if (toolUseID) {
        if (!toolUseGroups.has(toolUseID)) {
          toolUseGroups.set(toolUseID, {
            toolUse: null,
            preHooks: [],
            toolResult: null,
            postHooks: []
          })
        }
        toolUseGroups.get(toolUseID)!.toolUse = message
      }
    }
    
    // 处理 PreToolUse hooks
    if (isHookAttachmentMessage(message) && message.attachment.hookEvent === 'PreToolUse') {
      const group = toolUseGroups.get(message.attachment.toolUseID)
      if (group) group.preHooks.push(message)
    }
    
    // 处理 tool_result
    if (message.type === 'user' && message.message.content[0]?.type === 'tool_result') {
      const group = toolUseGroups.get(message.message.content[0].tool_use_id)
      if (group) group.toolResult = message
    }
    
    // 处理 PostToolUse hooks
    if (isHookAttachmentMessage(message) && message.attachment.hookEvent === 'PostToolUse') {
      const group = toolUseGroups.get(message.attachment.toolUseID)
      if (group) group.postHooks.push(message)
    }
  }
  
  // 第二遍：按正确顺序重建消息列表
  const result: typeof messages = []
  const processedToolUses = new Set<string>()
  
  for (const message of messages) {
    if (isToolUseRequestMessage(message)) {
      const toolUseID = message.message.content[0]?.id
      if (toolUseID && !processedToolUses.has(toolUseID)) {
        processedToolUses.add(toolUseID)
        const group = toolUseGroups.get(toolUseID)
        if (group && group.toolUse) {
          // 输出顺序：tool_use, pre hooks, tool_result, post hooks
          result.push(group.toolUse)
          result.push(...group.preHooks)
          if (group.toolResult) result.push(group.toolResult)
          result.push(...group.postHooks)
        }
      }
      continue
    }
    
    // 非工具调用组消息直接添加
    if (!isHookAttachmentMessage(message) && 
        !(message.type === 'user' && message.message.content[0]?.type === 'tool_result')) {
      result.push(message)
    }
  }
  
  return result
}
```

```typescript
export const HighlightedCode = memo(function HighlightedCode({
  code,
  filePath,
  width,
  dim = false,
}: Props) {
  const ref = useRef(null)
  const [measuredWidth, setMeasuredWidth] = useState(width || 80)
  const [theme] = useTheme()
  const settings = useSettings()
  
  // 获取 ColorFile 实例（懒加载）
  const colorFile = useMemo(() => {
    if (settings.syntaxHighlightingDisabled) return null
    
    const ColorFile = expectColorFile()
    if (!ColorFile) return null
    
    return new ColorFile(code, filePath)
  }, [code, filePath, settings.syntaxHighlightingDisabled])
  
  // 测量容器宽度
  useEffect(() => {
    if (!width && ref.current) {
      const { width: elementWidth } = measureElement(ref.current)
      if (elementWidth > 0) {
        setMeasuredWidth(elementWidth - 2)
      }
    }
  }, [width])
  
  // 渲染高亮行
  const lines = useMemo(() => {
    if (colorFile === null) return null
    return colorFile.render(theme, measuredWidth, dim)
  }, [colorFile, theme, measuredWidth, dim])
  
  // 计算行号宽度
  const gutterWidth = useMemo(() => {
    if (!isFullscreenEnvEnabled()) return 0
    const lineCount = countCharInString(code, '\n') + 1
    return lineCount.toString().length + 2
  }, [code])
  
  return (
    <Box ref={ref}>
      {lines ? (
        <Box flexDirection="column">
          {lines.map((line, i) => 
            gutterWidth > 0 ? (
              <CodeLine key={i} line={line} gutterWidth={gutterWidth} />
            ) : (
              <Text key={i}><Ansi>{line}</Ansi></Text>
            )
          )}
        </Box>
      ) : (
        <HighlightedCodeFallback 
          code={code} 
          filePath={filePath} 
          dim={dim} 
          skipColoring={settings.syntaxHighlightingDisabled} 
        />
      )}
    </Box>
  )
})
```

```typescript
export function UserToolResultMessage({
  param,

  progressMessagesForMessage,
  tools,
  verbose,
  width,
  isTranscriptMode,
}: Props) {
  // 获取对应的工具定义
  const toolUse = useGetToolFromMessages(param.tool_use_id, tools, lookups)
  if (!toolUse) return null
  
  // 用户取消
  if (typeof param.content === 'string' && param.content.startsWith(CANCEL_MESSAGE)) {
    return <UserToolCanceledMessage />
  }
  
  // 用户拒绝
  if ((typeof param.content === 'string' && param.content.startsWith(REJECT_MESSAGE)) ||

        input={toolUse.toolUse.input}
        tool={toolUse.tool}
        tools={tools}
        verbose={verbose}
        isTranscriptMode={isTranscriptMode}
      />

    return (
      <UserToolErrorMessage
        tool={toolUse.tool}
        tools={tools}
        param={param}
        verbose={verbose}
        isTranscriptMode={isTranscriptMode}
      />
    )
  }
  
  // 成功结果
  return (
    <UserToolSuccessMessage
      message={message}
      lookups={lookups}
      toolUseID={toolUse.toolUse.id}
      tool={toolUse.tool}
      tools={tools}
      verbose={verbose}
```

### 7.2 消息类型识别

#### 用户消息

- **普通文本** - 用户输入的问题或指令

#### 系统消息

- **压缩边界** - 历史对话被压缩的标记
- **API 错误** - 红色错误信息
- **本地命令输出** - 斜杠命令的输出

```typescript
import { marked } from 'marked'
import { Box, Text } from 'ink'
import React from 'react'

export function Markdown({ children }: { children: string }) {
  const tokens = React.useMemo(() => marked.lexer(children), [children])
  
  return (
    <Box flexDirection="column">
      {tokens.map((token, i) => (
        <TokenRenderer key={i} token={token} />
      ))}
    </Box>
  )
}

function TokenRenderer({ token }: { token: marked.Token }) {
  switch (token.type) {
    case 'paragraph':
      return <Text>{token.text}</Text>
    case 'heading':
      return <Text bold>{token.text}</Text>
    case 'code':
      return <CodeBlock code={token.text} lang={token.lang} />
    // ... 更多类型
    default:
      return <Text>{token.raw}</Text>
  }
}
```

### 8.3 实现流式渲染

#### 步骤 5：流式文本组件

```typescript
// components/StreamingText.tsx

import React, { useRef, useEffect } from 'react'
import { Markdown } from './Markdown'

export function StreamingText({ text }: { text: string }) {
  const prevLengthRef = useRef(0)
  const stablePrefixRef = useRef('')
  
  // 检测内容是否被重置
  if (text.length < prevLengthRef.current) {
    stablePrefixRef.current = ''
  }
  prevLengthRef.current = text.length
  
  // 找到最后一个完整的段落
  const lastNewline = text.lastIndexOf('\n\n')
  const stablePart = lastNewline > 0 ? text.slice(0, lastNewline) : ''
  const streamingPart = lastNewline > 0 ? text.slice(lastNewline) : text
  
  // 更新稳定部分
  if (stablePart.length > stablePrefixRef.current.length) {
    stablePrefixRef.current = stablePart
  }
  
  return (
    <Box flexDirection="column">
      {stablePrefixRef.current && (
        <Markdown>{stablePrefixRef.current}</Markdown>
      )}
      {streamingPart && (
        <Text dimColor>{streamingPart}</Text>
      )}
    </Box>
  )
}
```

```typescript
// components/VirtualMessageList.tsx

import React, { useRef, useState, useEffect } from 'react'
import { Box } from 'ink'

const ITEM_HEIGHT = 3  // 预估高度
const OVERSCAN = 5     // 额外渲染的项数

export function VirtualMessageList({
  messages,
  height,
  renderItem,
}: {
  messages: Message[]
  height: number
  renderItem: (message: Message, index: number) => React.ReactNode
}) {
  const [scrollTop, setScrollTop] = useState(0)
  
  const startIndex = Math.max(0, Math.floor(scrollTop / ITEM_HEIGHT) - OVERSCAN)
  const endIndex = Math.min(
    messages.length,
    Math.ceil((scrollTop + height) / ITEM_HEIGHT) + OVERSCAN
  )
  
  const visibleMessages = messages.slice(startIndex, endIndex)
  const offsetY = startIndex * ITEM_HEIGHT
  
  return (
    <Box flexDirection="column" height={height}>
      <Box height={offsetY} />
      {visibleMessages.map((msg, i) => renderItem(msg, startIndex + i))}
    </Box>
  )
}
```

```typescript
// App.tsx

import React, { useState, useEffect } from 'react'
import { Box, Text, useInput } from 'ink'
import { Messages } from './components/Messages'
import { StreamingText } from './components/StreamingText'
import { createUserMessage, createAssistantMessage } from './utils/messages'

export function App() {
  const [messages, setMessages] = useState<Message[]>([])
  const [streamingText, setStreamingText] = useState<string>('')
  const [isStreaming, setIsStreaming] = useState(false)
  
  // 模拟流式响应
  const simulateStream = async (text: string) => {
    setIsStreaming(true)
    setStreamingText('')
    
    for (const char of text) {
      setStreamingText(prev => prev + char)
      await new Promise(resolve => setTimeout(resolve, 10))
    }
    
    setMessages(prev => [...prev, createAssistantMessage(text)])
    setStreamingText('')
    setIsStreaming(false)
  }
  
  // 处理用户输入
  useInput((input, key) => {
    if (key.return && !isStreaming) {
      const userMsg = createUserMessage(input)
      setMessages(prev => [...prev, userMsg])
      simulateStream(`You said: "${input}"`)
    }
  })
  
  return (
    <Box flexDirection="column" height="100%">
      <Box flexGrow={1}>
        <Messages messages={messages} />
        {streamingText && <StreamingText text={streamingText} />}
      </Box>
      
      <Box>
        <Text dimColor>
          {isStreaming ? 'Thinking...' : 'Type a message and press Enter'}
        </Text>
      </Box>
    </Box>
  )
}
```
