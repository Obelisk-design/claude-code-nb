## 1. 终端检测概述

### 1.1 为什么需要终端能力检测

终端能力检测是 CLI 应用与终端交互的基础设施，主要解决以下问题：

1. **环境适配** - 不同终端支持的特性不同（颜色深度、鼠标事件、同步更新等）
2. **特性启用** - 根据终端能力动态启用/禁用高级功能
3. **兼容性处理** - 处理旧终端、SSH 环境、终端复用器（tmux/screen）的特殊情况
4. **性能优化** - 在支持的终端上使用更高效的渲染方式

### 3.2 CSI序列生成器（csi.ts）

**CSI序列结构**：

```
CSI序列 = ESC + '[' + 参数 + 最终字节
例如: \x1b[31m  (红色前景)
     \x1b[2J    (清除屏幕)
     \x1b[?25h  (显示光标，带私有模式前缀)
```

**CSI生成函数**（行45-51）：

```typescript
export function csi(...args: (string | number)[]): string {
  if (args.length === 0) return CSI_PREFIX  // "\x1b["
  if (args.length === 1) return `${CSI_PREFIX}${args[0]}`  // 单参数直接拼接
  const params = args.slice(0, -1)
  const final = args[args.length - 1]
  return `${CSI_PREFIX}${params.join(SEP)}${final}`  // 多参数用分号连接
}
```

**常用CSI命令常量**（行56-99）：

```typescript

  return csi(2, 'J')

  if (n <= 0) return ''
  let result = ''
  for (let i = 0; i < n; i++) {
    result += ERASE_LINE  // CSI 2 K
    if (i < n - 1) result += cursorUp(1)

  return result
```

```typescript
export function applySGR(paramStr: string, style: TextStyle): TextStyle {
  const params = parseParams(paramStr)
  let s = { ...style }
  let i = 0
  
  while (i < params.length) {
    const code = p.value ?? 0
    
    // 基础样式
    if (code === 0) s = defaultStyle()         // 重置
    if (code === 1) s.bold = true
    if (code === 3) s.italic = true
    if (code === 4) s.underline = 'single'
    
    // 16色前景（30-37）
    if (code >= 30 && code <= 37) {
      s.fg = { type: 'named', name: NAMED_COLORS[code - 30]! }
    }
    
    // 256色/Truecolor前景（38）
    if (code === 38) {
      const c = parseExtendedColor(params, i)
      if (c) {
        s.fg = 'index' in c
          ? { type: 'indexed', index: c.index }
          : { type: 'rgb', ...c }
      }
    }
    
    // ... 类似处理背景（40-47, 48）
  }
  return s
}
```

```typescript
function parseExtendedColor(params: Param[], idx: number) {
  // 冒号分隔格式: 38:5:N 或 38:2:R:G:B
  if (p.colon && p.subparams.length >= 1) {
    if (p.subparams[0] === 5) return { index: p.subparams[1] }
    if (p.subparams[0] === 2) return { r, g, b }  // RGB组件
  }
  
  // 分号分隔格式: 38;5;N 或 38;2;R;G;B
  if (next.value === 5) return { index: params[idx + 2].value }
  if (next.value === 2) return { r, g, b }
}
```

```typescript
export const colorize = (str: string, color: string | undefined, type: ColorType): string => {
  if (!color) return str
  
  // ANSI 16色: ansi:red, ansi:green
  if (color.startsWith('ansi:')) {
    const value = color.substring('ansi:'.length)
    return type === 'foreground' ? chalk.red(str) : chalk.bgRed(str)
  }
  
  // Hex颜色: #ff8c00
  if (color.startsWith('#')) {
    return type === 'foreground'
      ? chalk.hex(color)(str)
      : chalk.bgHex(color)(str)
  }
  
  // 256色: ansi256(215)
  if (color.startsWith('ansi256')) {
    const value = Number(matches[1])
    return chalk.ansi256(value)(str)
  }
  
  // RGB颜色: rgb(255,140,0)
  if (color.startsWith('rgb')) {
    const [r, g, b] = matches
    return chalk.rgb(r, g, b)(str)
  }
}
```

```typescript
function parseMouseEvent(s: string): ParsedMouse | null {
  const match = SGR_MOUSE_RE.exec(s)
  if (!match) return null
  const button = parseInt(match[1]!, 10)
  
  // 滚轮事件(bit 6)返回null，由键绑定系统处理
  if ((button & 0x40) !== 0) return null
  
  return {
    kind: 'mouse',
    button,
    action: match[4] === 'M' ? 'press' : 'release',
    col: parseInt(match[2]!, 10),
    row: parseInt(match[3]!, 10),
    sequence: s,
  }
}
```

### 5.3 X10与SGR格式对比

**X10格式（旧）**：

```
\x1b[M Cb Cx Cy
- 固定6字节
- Cb/Cx/Cy 是原始字节（+32偏移）
- 坐标限制：0-223（超过无法编码）
- 无法区分按下/释放（都是'M'）
```

**SGR格式（新）**：

```
\x1b[<button;col;row M 或 m
- 可解析的数字参数
- M=按下, m=释放
- 支持任意坐标
- 包含修饰键信息
```

```typescript
function detectTerminal(): string | null {
  // Cursor IDE
  if (process.env.CURSOR_TRACE_ID) return 'cursor'
  
  // VS Code系列（检测Git Askpass路径）
  if (process.env.VSCODE_GIT_ASKPASS_MAIN?.includes('cursor')) return 'cursor'
  if (process.env.VSCODE_GIT_ASKPASS_MAIN?.includes('windsurf')) return 'windsurf'
  
  // macOS bundle ID检测
  const bundleId = process.env.__CFBundleIdentifier?.toLowerCase()
  if (bundleId?.includes('vscodium')) return 'codium'
  if (bundleId?.includes('windsurf')) return 'windsurf'
  
  // JetBrains IDEs
  for (const ide of JETBRAINS_IDES) {
    if (bundleId?.includes(ide)) return ide
  }
  
  // Linux/Windows JetBrains
  if (process.env.TERMINAL_EMULATOR === 'JetBrains-JediTerm') return 'pycharm'
  
  // TERM环境变量检测
  if (process.env.TERM === 'xterm-ghostty') return 'ghostty'
  if (process.env.TERM?.includes('kitty')) return 'kitty'
  
  // TERM_PROGRAM（macOS）
  if (process.env.TERM_PROGRAM) return process.env.TERM_PROGRAM
  
  // 终端复用器
  if (process.env.TMUX) return 'tmux'
  if (process.env.STY) return 'screen'
  
  // Linux终端环境变量
  if (process.env.KONSOLE_VERSION) return 'konsole'
  if (process.env.GNOME_TERMINAL_SERVICE) return 'gnome-terminal'
  if (process.env.KITTY_WINDOW_ID) return 'kitty'
  
  // Windows终端
  if (process.env.WT_SESSION) return 'windows-terminal'
  if (process.env.ConEmuANSI) return 'conemu'
  
  // WSL
  if (process.env.WSL_DISTRO_NAME) return `wsl-${process.env.WSL_DISTRO_NAME}`
  
  // SSH会话
  if (isSSHSession()) return 'ssh-session'
  
  // 最终回退
  if (process.env.TERM) return process.env.TERM
  if (!process.stdout.isTTY) return 'non-interactive'
  
  return null
}
```

```typescript
// terminal-query.ts
import { csi } from './minimal-csi'

interface TerminalQuery<T> {
  request: string
  match: (response: any) => response is T
}

// DECRQM: 查询DEC模式状态
export function decrqm(mode: number): TerminalQuery<{ mode: number; status: number }> {
  return {
    request: csi(`?${mode}$p`),
    match: (r) => r.type === 'decrpm' && r.mode === mode,
  }
}

// DA1: 设备属性（通用哨兵）
export function da1(): TerminalQuery<{ params: number[] }> {
  return {
    request: csi('', 'c'),
    match: (r) => r.type === 'da1',
  }
}

// 简化的查询器
class SimpleQuerier {
  private pending: Array<{
    match: (r: any) => boolean
    resolve: (r: any) => void
  }> = []
  
  send(query: TerminalQuery<any>): Promise<any> {
    return new Promise(resolve => {
      this.pending.push({ match: query.match, resolve })
      process.stdout.write(query.request)
    })
  }
  
  // DA1作为哨兵，响应表示所有先前查询完成
  async flush(): Promise<void> {
    await this.send(da1())
  }
  
  onResponse(response: any): void {
    const idx = this.pending.findIndex(p => p.match(response))
    if (idx !== -1) {
      const [handler] = this.pending.splice(idx, 1)
      handler.resolve(response)
    }
  }
}
```

### 7.4 鼠标事件解析

```typescript
// mouse-parser.ts
const SGR_MOUSE_RE = /^\x1b\[<(\d+);(\d+);(\d+)([Mm])$/

interface MouseEvent {
  button: number
  action: 'press' | 'release'
  col: number
  row: number
}

export function parseMouse(input: string): MouseEvent | null {
  const match = SGR_MOUSE_RE.exec(input)
  if (!match) return null
  
  const button = parseInt(match[1], 10)
  
  // 滚轮事件(bit 6)单独处理
  if (button & 0x40) {
    const direction = (button & 1) ? 'down' : 'up'
    return { button: 'wheel', action: direction, col: 0, row: 0 }
  }
  
  return {
    button: button & 0x03,  // 0=左, 1=中, 2=右
    action: match[4] === 'M' ? 'press' : 'release',
    col: parseInt(match[2], 10),
    row: parseInt(match[3], 10),
  }
}
```

### 7.5 输入流分词器

```typescript
// tokenizer.ts
const ESC = 0x1b
const CSI_FINAL_START = 0x40  // '@'
const CSI_FINAL_END = 0x7e    // '~'

type Token = { type: 'text'; value: string } | { type: 'sequence'; value: string }

export function tokenize(input: string): Token[] {
  const tokens: Token[] = []
  let i = 0
  let textStart = 0
  
  while (i < input.length) {
    if (input.charCodeAt(i) === ESC) {
      // 输出之前的文本
      if (i > textStart) {
        tokens.push({ type: 'text', value: input.slice(textStart, i) })
      }
      
      // 找到转义序列结束
      const seqStart = i
      i++  // 跳过ESC
      
      if (input.charCodeAt(i) === 0x5b) {  // '[' = CSI
        i++
        // 查找最终字节
        while (i < input.length) {
          const c = input.charCodeAt(i)
          if (c >= CSI_FINAL_START && c <= CSI_FINAL_END) {
            i++
            break
          }
          i++
        }
      } else if (input.charCodeAt(i) === 0x5d) {  // ']' = OSC
        i++
        // OSC以BEL或ST结束
        while (i < input.length && input.charCodeAt(i) !== 0x07) {
          if (input.charCodeAt(i) === ESC && input.charCodeAt(i+1) === 0x5c) {
            i += 2
            break
          }
          i++
        }
      } else {
        // 简单ESC序列（2字符）
        i++
      }
      
      tokens.push({ type: 'sequence', value: input.slice(seqStart, i) })
      textStart = i
    } else {
      i++
    }
  }
  
  // 剩余文本
  if (i > textStart) {
    tokens.push({ type: 'text', value: input.slice(textStart, i) })
  }
  
  return tokens
}
```

### 7.6 完整示例应用

```typescript
// app.ts
import * as CSI from './minimal-csi'
import { detectTerminal, supportsTruecolor, supportsMouse } from './terminal-detect'
import { parseMouse } from './mouse-parser'
import { tokenize } from './tokenizer'

class TerminalApp {
  private terminalType: string
  private mouseEnabled = false
  
  constructor() {
    this.terminalType = detectTerminal()
    this.setup()
  }
  
  private setup(): void {
    // 启用鼠标跟踪（如果支持）
    if (supportsMouse()) {
      process.stdout.write(CSI.decset(1000) + CSI.decset(1006))
      this.mouseEnabled = true
    }
    
    // 进入备用屏幕
    process.stdout.write(CSI.decset(1049))
    
    // 设置原始模式读取输入
    process.stdin.setRawMode(true)
    process.stdin.on('data', this.handleInput.bind(this))
  }
  
  private handleInput(data: Buffer): void {
    const input = data.toString()
    const tokens = tokenize(input)
    
    for (const token of tokens) {
      if (token.type === 'sequence') {
        const mouse = parseMouse(token.value)
        if (mouse) {
          this.handleMouse(mouse)
        }
      } else if (token.type === 'text') {
        // 键盘输入
        if (token.value === '\x03') {  // Ctrl+C
          this.cleanup()
          process.exit()
        }
      }
    }
  }
  
  private handleMouse(event: MouseEvent): void {
    // 清除行并显示鼠标信息
    process.stdout.write(CSI.eraseLine() + CSI.cursorTo(1))
    process.stdout.write(`Mouse: button=${event.button} at (${event.col}, ${event.row})`)
  }
  
  private cleanup(): void {
    if (this.mouseEnabled) {
      process.stdout.write(CSI.decreset(1006) + CSI.decreset(1000))
    }
    process.stdout.write(CSI.decreset(1049))  // 退出备用屏幕
    process.stdin.setRawMode(false)
  }
}

// 运行
const app = new TerminalApp()
```
