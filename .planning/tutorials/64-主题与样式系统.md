## 1. 主题系统概述

Claude Code CLI 的主题系统分为两个独立但互补的部分：

### 1.1 视觉主题（Visual Theme）

**位置：** `utils/theme.ts`

**功能：** 控制终端 UI 的颜色方案，包括文本颜色、边框颜色、背景色等。

**支持的视觉主题：**
- `dark` - 深色主题
- `light` - 浅色主题
- `dark-daltonized` - 色盲友好的深色主题
- `light-daltonized` - 色盲友好的浅色主题
- `dark-ansi` - 仅使用 ANSI 16 色的深色主题
- `light-ansi` - 仅使用 ANSI 16 色的浅色主题
- `auto` - 自动跟随终端主题（可选）

### 1.2 输出样式（Output Style）

**位置：** `constants/outputStyles.ts`、`outputStyles/loadOutputStylesDir.ts`

**功能：** 控制 Claude 的响应风格和交互模式。

**内置输出样式：**
- `default` - 默认高效模式
- `Explanatory` - 解释模式，提供教育性见解
- `Learning` - 学习模式，邀请用户动手实践

### 2.4 ThemePicker 组件详解

**文件位置：** `components/ThemePicker.tsx`

**核心功能：**
1. 显示主题选项列表
2. 提供实时预览
3. 支持语法高亮切换

```typescript
export function ThemePicker({
  onThemeSelect,
  showIntroText = false,
  helpText = '',
  showHelpTextBelow = false,
  hideEscToCancel = false,
  skipExitHandling = false,
  onCancel,

  // 主题选项

    ...(feature('AUTO_THEME') ? [{
      label: "Auto (match terminal)",
      value: "auto" as const

    { label: "Light mode", value: "light" },
    { label: "Dark mode (colorblind-friendly)", value: "dark-daltonized" },
    { label: "Light mode (colorblind-friendly)", value: "light-daltonized" },
    { label: "Dark mode (ANSI colors only)", value: "dark-ansi" },
    { label: "Light mode (ANSI colors only)", value: "light-ansi" },

  return (
    <Box flexDirection="column" gap={1}>
      {/* 标题 */}

      {/* 选择器 */}
      <Select
        options={themeOptions}
        onFocus={setting => {
          setPreviewTheme(setting as ThemeSetting)
        }}
        onChange={setting => {
          savePreview()
          onThemeSelect(setting as ThemeSetting)
        }}
        onCancel={skipExitHandling ? () => {
          cancelPreview()
          onCancel?.()

          cancelPreview()

              " }"
            ]
          }}
          filePath="demo.js"
          width={columns}
        />
```

```typescript
export function getTheme(themeName: ThemeName): Theme {
  switch (themeName) {
    case 'light':
      return lightTheme
    case 'light-ansi':
      return lightAnsiTheme
    case 'dark-ansi':
      return darkAnsiTheme
    case 'light-daltonized':
      return lightDaltonizedTheme
    case 'dark-daltonized':
      return darkDaltonizedTheme
    default:
      return darkTheme
  }
}
```

### 3.5 系统主题检测

**文件位置：** `utils/systemTheme.ts`

```typescript
export type SystemTheme = 'dark' | 'light'

// 获取系统主题（带缓存）
export function getSystemThemeName(): SystemTheme {
  if (cachedSystemTheme === undefined) {
    cachedSystemTheme = detectFromColorFgBg() ?? 'dark'
  }
  return cachedSystemTheme
}

// 解析 OSC 11 颜色响应
export function themeFromOscColor(data: string): SystemTheme | undefined {
  const rgb = parseOscRgb(data)
  if (!rgb) return undefined

  // ITU-R BT.709 相对亮度计算
  const luminance = 0.2126 * rgb.r + 0.7152 * rgb.g + 0.0722 * rgb.b
  return luminance > 0.5 ? 'light' : 'dark'
}

// 从 $COLORFGBG 环境变量检测（同步）
function detectFromColorFgBg(): SystemTheme | undefined {
  const colorfgbg = process.env['COLORFGBG']
  if (!colorfgbg) return undefined

  const parts = colorfgbg.split(';')
  const bg = parts[parts.length - 1]
  const bgNum = Number(bg)

  // 0-6 和 8 是深色 ANSI 颜色，7 和 9-15 是浅色
  return bgNum <= 6 || bgNum === 8 ? 'dark' : 'light'
}
```

### 4.3 输出样式加载流程

**文件位置：** `outputStyles/loadOutputStylesDir.ts`

```typescript
export const getOutputStyleDirStyles = memoize(
  async (cwd: string): Promise<OutputStyleConfig[]> => {
    try {
      // 1. 加载 Markdown 文件
      const markdownFiles = await loadMarkdownFilesForSubdir(
        'output-styles',
        cwd,
      )

      // 2. 解析每个文件
      const styles = markdownFiles
        .map(({ filePath, frontmatter, content, source }) => {
          try {
            const fileName = basename(filePath)
            const styleName = fileName.replace(/\.md$/, '')

            // 从 frontmatter 获取配置
            const name = (frontmatter['name'] || styleName) as string
            const description =
              coerceDescriptionToString(frontmatter['description'], styleName) ??
              extractDescriptionFromMarkdown(content, `Custom ${styleName} output style`)

            // 解析 keep-coding-instructions 标志
            const keepCodingInstructionsRaw = frontmatter['keep-coding-instructions']
            const keepCodingInstructions =
              keepCodingInstructionsRaw === true || keepCodingInstructionsRaw === 'true'

            return {
              name,
              description,
              prompt: content.trim(),
              source,
              keepCodingInstructions,
            }
          } catch (error) {
            logError(error)
            return null
          }
        })
        .filter(style => style !== null)

      return styles
    } catch (error) {
      logError(error)
      return []
    }
  },
)
```

### 4.4 输出样式合并优先级

```typescript
export const getAllOutputStyles = memoize(async function getAllOutputStyles(
  cwd: string,
): Promise<{ [styleName: string]: OutputStyleConfig | null }> {
  const customStyles = await getOutputStyleDirStyles(cwd)
  const pluginStyles = await loadPluginOutputStyles()

  // 从内置模式开始
  const allStyles = {
    ...OUTPUT_STYLE_CONFIG,
  }

  // 按优先级分组（从低到高）
  const managedStyles = customStyles.filter(style => style.source === 'policySettings')
  const userStyles = customStyles.filter(style => style.source === 'userSettings')
  const projectStyles = customStyles.filter(style => style.source === 'projectSettings')

  // 按优先级顺序添加：built-in < plugin < managed < user < project
  const styleGroups = [pluginStyles, userStyles, projectStyles, managedStyles]

  for (const styles of styleGroups) {
    for (const style of styles) {
      allStyles[style.name] = {
        name: style.name,
        description: style.description,
        prompt: style.prompt,
        source: style.source,
        keepCodingInstructions: style.keepCodingInstructions,
        forceForPlugin: style.forceForPlugin,
      }
    }
  }

  return allStyles
})
```

**优先级顺序：**

```
1. built-in (内置样式)
   ↓ 被覆盖
2. plugin (插件样式)
   ↓ 被覆盖
3. userSettings (~/.claude/output-styles/)
   ↓ 被覆盖
4. projectSettings (.claude/output-styles/)
   ↓ 被覆盖
5. policySettings (管理策略)
```

```typescript
// theme-context.ts
import React, { createContext, useContext, useState, useMemo } from 'react'

type ThemeMode = 'light' | 'dark'

interface ThemeContextValue {
  mode: ThemeMode
  setMode: (mode: ThemeMode) => void
}

const ThemeContext = createContext<ThemeContextValue>({
  mode: 'dark',
  setMode: () => {},
})

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [mode, setMode] = useState<ThemeMode>('dark')

  const value = useMemo(() => ({
    mode,
    setMode,
  }), [mode])

  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  )
}

export function useTheme() {
  return useContext(ThemeContext)
}
```

```typescript
// ThemedText.tsx
import React, { useContext } from 'react'
import Text from './Text'
import { themes, Theme } from './themes'
import { useTheme } from './theme-context'

interface Props {
  color?: keyof Theme | string
  children: React.ReactNode
}

function resolveColor(color: string | undefined, theme: Theme): string | undefined {
  if (!color) return undefined

  // 如果是主题键，解析它
  if (color in theme) {
    return theme[color as keyof Theme]
  }

  // 否则假设它是原始颜色值
  return color
}

export default function ThemedText({ color, children }: Props) {
  const { mode } = useTheme()
  const theme = themes[mode]
  const resolvedColor = resolveColor(color, theme)

  return (
    <Text color={resolvedColor}>
      {children}
    </Text>
  )
}
```

### 6.5 系统主题检测实现

```typescript
// useSystemTheme.ts
import { useState, useEffect } from 'react'

export function useSystemTheme(): 'dark' | 'light' {
  const [systemTheme, setSystemTheme] = useState<'dark' | 'light'>('dark')

  useEffect(() => {
    // 方法 1: 检查 prefers-color-scheme（浏览器）
    if (typeof window !== 'undefined' && window.matchMedia) {
      const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)')
      setSystemTheme(mediaQuery.matches ? 'dark' : 'light')

      const handler = (e: MediaQueryListEvent) => {
        setSystemTheme(e.matches ? 'dark' : 'light')
      }
      mediaQuery.addEventListener('change', handler)
      return () => mediaQuery.removeEventListener('change', handler)
    }

    // 方法 2: 检查 COLORFGBG 环境变量（终端）
    const colorfgbg = process.env.COLORFGBG
    if (colorfgbg) {
      const bg = colorfgbg.split(';').pop()
      if (bg) {
        const bgNum = parseInt(bg, 10)
        setSystemTheme(bgNum <= 6 || bgNum === 8 ? 'dark' : 'light')
      }
    }
  }, [])

  return systemTheme
}
```

### 6.6 主题切换命令实现

```typescript
// commands/theme.ts
import { Command } from '../command-types'

export const themeCommand: Command = {
  name: 'theme',
  description: 'Change the visual theme',
  async execute(args, context) {
    const subcommand = args[0]

    if (subcommand === 'list') {
      return {
        output: 'Available themes:\n' +
                '  dark    - Dark mode\n' +
                '  light   - Light mode\n' +
                '  auto    - Follow system'
      }
    }

    if (subcommand === 'set') {
      const themeName = args[1]
      if (!['dark', 'light', 'auto'].includes(themeName)) {
        return { error: `Unknown theme: ${themeName}` }
      }

      context.setTheme(themeName)
      return { output: `Theme set to ${themeName}` }
    }

    return {
      output: 'Usage: theme <list|set> [theme-name]'
    }
  }
}
```
