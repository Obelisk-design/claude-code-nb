# 沙箱隔离系统详解

**分析日期:** 2026-04-02

## 1. 沙箱设计目标

Claude Code CLI 的沙箱系统旨在为 Bash 命令执行提供操作系统级别的安全隔离，防止恶意命令对用户系统造成损害。

### 核心目标

**目标1：限制文件系统访问**
- 只允许在指定目录读写
- 阻止对敏感配置文件的修改（如 `.claude/settings.json`）
- 阻止对 Git 配置的修改（防止 hook 注入攻击）

**目标2：限制网络访问**
- 只允许访问白名单域名
- 阻止对内网服务的访问
- 支持 HTTP/SOCKS 代理配置

**目标3：跨平台支持**
- macOS：使用系统内置的 `sandbox-exec`
- Linux/WSL2：使用 `bubblewrap` + `socat`
- Windows/WSL1：不支持（技术限制）

**目标4：透明执行**
- 沙箱对命令执行透明封装
- 用户无需修改命令语法
- 沙箱失败时优雅降级（非强制模式）

### 安全边界说明

```
┌─────────────────────────────────────────────────────────────┐
│                     Claude Code CLI                          │
│  ┌─────────────────┐                                        │
│  │   Bash Tool     │ ──► 权限检查 ──► 沙箱封装 ──► 执行     │
│  └─────────────────┘                                        │
│                                                              │
│  沙箱不是安全边界：                                          │
│  - 权限系统才是真正的安全控制                                │
│  - excludedCommands 只是便利功能                             │
│  - 用户可通过 dangerouslyDisableSandbox 禁用                │
└─────────────────────────────────────────────────────────────┘
```

```typescript
/**
 * 解析 Claude Code 特有的路径模式
 * 
 * - `//path` → 绝对路径（从根开始）
 * - `/path`  → 相对于 settings 文件目录
 * - `~/path` → 传递给 sandbox-runtime 处理
 * - `./path` → 传递给 sandbox-runtime 处理
 */
export function resolvePathPatternForSandbox(
  pattern: string,
  source: SettingSource,
): string {
  if (pattern.startsWith('//')) {
    return pattern.slice(1)  // "//.aws/**" → "/.aws/**"
  }
  
  if (pattern.startsWith('/') && !pattern.startsWith('//')) {
    const root = getSettingsRootPathForSource(source)
    return resolve(root, pattern.slice(1))
  }
  
  return pattern  // 其他模式原样传递
}
```

**关键点：**
- Claude Code 有特殊的路径约定
- permission rules 和 sandbox.filesystem 设置使用不同的语义
- Fix #30067：sandbox.filesystem 中的 `/path` 是绝对路径，不是相对路径

### 2.4 配置转换函数（第172-381行）

这是最核心的函数，将 Claude Code settings 转换为 sandbox-runtime 配置。

```typescript
export function convertToSandboxRuntimeConfig(
  settings: SettingsJson,
): SandboxRuntimeConfig {
  const permissions = settings.permissions || {}
  
  // ========== 网络：从 WebFetch 规则提取域名 ==========
  const allowedDomains: string[] = []
  const deniedDomains: string[] = []
  
  if (shouldAllowManagedSandboxDomainsOnly()) {
    // 只使用 policySettings 中的域名（企业策略）
    const policySettings = getSettingsForSource('policySettings')
    for (const domain of policySettings?.sandbox?.network?.allowedDomains || []) {
      allowedDomains.push(domain)
    }
  } else {
    // 合并所有 settings 源的域名
    for (const domain of settings.sandbox?.network?.allowedDomains || []) {
      allowedDomains.push(domain)
    }
    for (const ruleString of permissions.allow || []) {
      const rule = permissionRuleValueFromString(ruleString)
      if (rule.toolName === WEB_FETCH_TOOL_NAME && 
          rule.ruleContent?.startsWith('domain:')) {
        allowedDomains.push(rule.ruleContent.substring('domain:'.length))
      }
    }
  }
  
  // ========== 文件系统：构建读写权限列表 ==========
  const allowWrite: string[] = ['.', getClaudeTempDir()]
  const denyWrite: string[] = []
  const denyRead: string[] = []
  
  // 安全加固：阻止修改 settings.json（防止沙箱逃逸）
  const settingsPaths = SETTING_SOURCES.map(source =>
    getSettingsFilePathForSource(source)
  ).filter((p): p is string => p !== undefined)
  denyWrite.push(...settingsPaths)
  
  // 安全加固：阻止修改 .claude/skills（特权级别文件）
  denyWrite.push(resolve(originalCwd, '.claude', 'skills'))
  
  // 安全加固：阻止 bare Git repo 植入攻击
  // (#29316: HEAD + objects/ + refs/ + hooks/ + config)
  const bareGitRepoFiles = ['HEAD', 'objects', 'refs', 'hooks', 'config']
  for (const dir of [originalCwd, cwd]) {
    for (const gitFile of bareGitRepoFiles) {
      const p = resolve(dir, gitFile)
      try {
        statSync(p)  // 存在则 deny（ro-bind）
        denyWrite.push(p)
      } catch {
        bareGitRepoScrubPaths.push(p)  // 不存在则事后清理
      }
    }
  }
  
  // Git worktree 支持：允许写入主 repo 的 .git
  if (worktreeMainRepoPath && worktreeMainRepoPath !== cwd) {
    allowWrite.push(worktreeMainRepoPath)
  }
  
  // ========== 返回配置 ==========
  return {
    network: {
      allowedDomains,
      deniedDomains,
      allowUnixSockets: settings.sandbox?.network?.allowUnixSockets,
      allowAllUnixSockets: settings.sandbox?.network?.allowAllUnixSockets,
      allowLocalBinding: settings.sandbox?.network?.allowLocalBinding,
      httpProxyPort: settings.sandbox?.network?.httpProxyPort,
      socksProxyPort: settings.sandbox?.network?.socksProxyPort,
    },
    filesystem: {
      denyRead,
      allowRead,
      allowWrite,
      denyWrite,
    },
    ignoreViolations: settings.sandbox?.ignoreViolations,
    enableWeakerNestedSandbox: settings.sandbox?.enableWeakerNestedSandbox,
    enableWeakerNetworkIsolation: settings.sandbox?.enableWeakerNetworkIsolation,
    ripgrep: ripgrepConfig,
  }
}
```

### 2.5 Git Worktree 检测（第422-445行）

```typescript
/**
 * 检测是否在 Git worktree 中，并解析主 repo 路径
 * 
 * worktree 的 .git 是文件（不是目录），内容如：
 * "gitdir: /path/to/main/repo/.git/worktrees/name"
 */
async function detectWorktreeMainRepoPath(cwd: string): Promise<string | null> {
  const gitPath = join(cwd, '.git')
  try {
    const gitContent = await readFile(gitPath, { encoding: 'utf8' })
    const gitdirMatch = gitContent.match(/^gitdir:\s*(.+)$/m)
    if (!gitdirMatch?.[1]) return null
    
    const gitdir = resolve(cwd, gitdirMatch[1].trim())
    // 提取主 repo 路径：/.git/worktrees/ 之前的部分
    const marker = `${sep}.git${sep}worktrees${sep}`
    const markerIndex = gitdir.lastIndexOf(marker)
    if (markerIndex > 0) {
      return gitdir.substring(0, markerIndex)
    }
    return null
  } catch {
    return null  // EISDIR = 不是 worktree
  }
}
```

### 2.7 初始化和配置刷新（第730-803行）

```typescript
async function initialize(sandboxAskCallback?: SandboxAskCallback): Promise<void> {
  if (initializationPromise) return initializationPromise
  
  if (!isSandboxingEnabled()) return
  
  initializationPromise = (async () => {
    try {
      // 解析 worktree 主 repo 路径（一次性）
      if (worktreeMainRepoPath === undefined) {
        worktreeMainRepoPath = await detectWorktreeMainRepoPath(getCwdState())
      }
      
      const settings = getSettings_DEPRECATED()
      const runtimeConfig = convertToSandboxRuntimeConfig(settings)
      
      await BaseSandboxManager.initialize(runtimeConfig, wrappedCallback)
      
      // 监听 settings 变化，动态更新配置
      settingsSubscriptionCleanup = settingsChangeDetector.subscribe(() => {
        const newConfig = convertToSandboxRuntimeConfig(getSettings_DEPRECATED())
        BaseSandboxManager.updateConfig(newConfig)
      })
    } catch (error) {
      initializationPromise = undefined
      logForDebugging(`Failed to initialize sandbox: ${errorMessage(error)}`)
    }
  })()
  
  return initializationPromise
}

// 同步刷新配置（避免竞态条件）
function refreshConfig(): void {
  if (!isSandboxingEnabled()) return
  const settings = getSettings_DEPRECATED()
  const newConfig = convertToSandboxRuntimeConfig(settings)
  BaseSandboxManager.updateConfig(newConfig)
}
```

## 3. macOS sandbox-exec

### 3.1 sandbox-exec 概述

macOS 内置的 Seatbelt 沙箱系统，通过 `sandbox-exec` 命令执行。

**优势：**
- 系统内置，无需额外安装
- 稳定可靠，Apple 官方维护
- 支持 Scheme 语法的 profile 配置

**执行流程：**

```
                    macOS sandbox-exec 流程
                    ┌───────────────────────────┐
                    │   1. 生成 profile (SBPL)   │
                    │   (version 1)              │
                    │   (deny default)           │
                    │   (allow file-read* ...)   │
                    │   (allow network* ...)     │
                    └───────────────────────────┘
                              │
                              ▼
                    ┌───────────────────────────┐
                    │   2. 写入临时 profile 文件  │
                    │   /tmp/claude-sandbox-XXX  │
                    └───────────────────────────┘
                              │
                              ▼
                    ┌───────────────────────────┐
                    │   3. 执行                  │
                    │   sandbox-exec -f profile │
                    │   /bin/bash -c "command"  │
                    └───────────────────────────┘
                              │
                              ▼
                    ┌───────────────────────────┐
                    │   4. 监控违规              │
                    │   日志监控进程             │
```

### 3.2 Profile 语法（SBPL）

```scheme
(version 1)                    ; 版本声明
(deny default)                 ; 默认拒绝所有
(deny process-exec)            ; 可选：阻止执行其他程序

; 文件系统规则
(allow file-read*              ; 允许读取
    (path "/Users/user/project")
    (path "/usr")
    (path "/bin"))
    
(allow file-write*             ; 允许写入
    (path "/Users/user/project")
    (path "/tmp/claude-xxx"))

; 网络规则
(allow network*                ; 允许网络
    (remote tcp "example.com:443"))
    
; Unix socket 规则
(allow network*                ; Unix socket
    (path "/tmp/mysocket"))
```

## 4. Linux bubblewrap

### 4.1 bubblewrap 概述

bubblewrap (`bwrap`) 是 Flatpak 项目开发的轻量级沙箱工具。

**依赖：**
- `bwrap` - 沙箱封装
- `socat` - 网络代理（用于域名过滤）

**安装：**
```bash
# Ubuntu/Debian
sudo apt install bubblewrap socat

# Fedora
sudo dnf install bubblewrap socat

# Arch Linux
sudo pacman -S bubblewrap socat
```

### 4.2 bwrap 参数详解

```
│  │  ┌─────────┐ │                                           │
```

### 4.4 Linux Glob 模式限制

bubblewrap 不支持 glob 模式路径匹配（如 `*.json`）。

```typescript
// 检测 glob 模式警告
function getLinuxGlobPatternWarnings(): string[] {
  if (platform !== 'linux' && platform !== 'wsl') return []
  
  const warnings: string[] = []
  for (const ruleString of permissions.allow || []) {
    const rule = permissionRuleValueFromString(ruleString)
    if (rule.ruleContent && hasGlobs(rule.ruleContent)) {
      warnings.push(ruleString)  // 如 "Bash(./node_modules/**)"
    }
  }
  return warnings
}

// hasGlobs 检查（排除合法的 /** 后缀）
const hasGlobs = (path: string): boolean => {
  const stripped = path.replace(/\/\*\*$/, '')  // 移除 /** 后缀
  return /[*?[\]]/.test(stripped)  // 检查其他 glob 字符
}
```

```typescript
  getSettingsFilePathForSource(source)

denyWrite.push(resolve(cwd, '.claude', 'skills'))

// 3. Bare Git repo 文件（防止 hook 注入）
// CVE #29316: 攻击者可植入 HEAD + objects/ + refs/ + config
// Git 的 is_git_directory() 会将 cwd 视为 bare repo
// 如果 config 中有 core.fsmonitor，会在 unsandboxed git 中触发 hook
const bareGitRepoFiles = ['HEAD', 'objects', 'refs', 'hooks', 'config']
for (const gitFile of bareGitRepoFiles) {
  const p = resolve(dir, gitFile)
  try {
    statSync(p)
    denyWrite.push(p)  // 存在：ro-bind 保持原样
  } catch {
    bareGitRepoScrubPaths.push(p)  // 不存在：事后清理
  }
}
```

```typescript
// src/sandbox/platform.ts
export type Platform = 'macos' | 'linux' | 'wsl' | 'windows' | 'unknown'

export function detectPlatform(): Platform {
  if (process.platform === 'darwin') return 'macos'
  if (process.platform === 'win32') return 'windows'
  
  if (process.platform === 'linux') {
    try {
      const procVersion = fs.readFileSync('/proc/version', 'utf8')
      if (procVersion.toLowerCase().includes('microsoft')) {
        return 'wsl'
      }
    } catch {}
    return 'linux'
  }
  
  return 'unknown'
}

export function checkDependencies(platform: Platform): {
  errors: string[]
  warnings: string[]
} {
  if (platform === 'macos') {
    // sandbox-exec 内置
    return { errors: [], warnings: [] }
  }
  
  if (platform === 'linux' || platform === 'wsl') {
    const errors: string[] = []
    if (!commandExists('bwrap')) errors.push('Missing: bwrap (bubblewrap)')
    if (!commandExists('socat')) errors.push('Missing: socat')
    return { errors, warnings: [] }
  }
  
  return { errors: ['Platform not supported'], warnings: [] }
}
```

### 8.3 步骤3：macOS Profile 生成

```typescript
// src/sandbox/macos.ts
export function generateSandboxProfile(config: SandboxConfig): string {
  const lines: string[] = [
    '(version 1)',
    '(deny default)',
  ]
  
  // 文件系统：允许读取
  for (const path of config.filesystem.allowRead) {
    lines.push(`(allow file-read* (path "${path}"))`)
  }
  for (const path of config.filesystem.allowWrite) {
    lines.push(`(allow file-read* (path "${path}"))`)  // 写入也需要读取权限
  }
  
  // 文件系统：允许写入
  for (const path of config.filesystem.allowWrite) {
    lines.push(`(allow file-write* (path "${path}"))`)
  }
  
  // 网络：允许域名
  for (const domain of config.network.allowedDomains) {
    lines.push(`(allow network-outbound (remote tcp "${domain}"))`)
    lines.push(`(allow network-outbound (remote tcp "${domain}:443"))`)
  }
  
  // Unix socket
  if (config.network.allowUnixSockets) {
    for (const socket of config.network.allowUnixSockets) {
      lines.push(`(allow network-outbound (path "${socket}"))`)
    }
  }
  
  return lines.join('\n')
}

export async function wrapMacOSCommand(
  command: string,
  config: SandboxConfig,
  shell: string
): Promise<string> {
  const profile = generateSandboxProfile(config)
  const profilePath = `/tmp/sandbox-profile-${Date.now()}.sb`
  
  await fs.writeFile(profilePath, profile)
  
  return `sandbox-exec -f ${profilePath} ${shell} -c '${command}'`
}
```

```typescript
// src/sandbox/linux.ts
export function generateBwrapArgs(config: SandboxConfig): string[] {
  const args: string[] = []
  
  // 系统目录只读挂载
  args.push('--ro-bind', '/usr', '/usr')
  args.push('--ro-bind', '/bin', '/bin')
  args.push('--ro-bind', '/lib', '/lib')
  args.push('--ro-bind', '/lib64', '/lib64')
  args.push('--ro-bind', '/etc', '/etc')
  
  // 工作目录读写挂载
  for (const path of config.filesystem.allowWrite) {
    args.push('--bind', path, path)
  }
  
  // 阻止写入路径
  for (const path of config.filesystem.denyWrite) {
    args.push('--bind', '/dev/null', path)  // 或 --ro-bind 保持原样
  }
  
  // 阻止读取路径
  for (const path of config.filesystem.denyRead) {
    args.push('--bind', '/dev/null', path)
  }
  
  // 网络隔离
  args.push('--unshare-net')
  
  // 新 dev 目录
  args.push('--dev', '/dev')
  
  // Proc 目录
  args.push('--proc', '/proc')
  
  return args
}

export async function wrapLinuxCommand(
  command: string,
  config: SandboxConfig,
  shell: string,
  proxyPort?: number
): Promise<string> {
  const bwrapArgs = generateBwrapArgs(config)
  
  // 设置代理环境变量（如果配置了）
  const envPrefix = proxyPort 
    ? `http_proxy=http://127.0.0.1:${proxyPort} ` 
    : ''
  
  return `${envPrefix}bwrap ${bwrapArgs.join(' ')} ${shell} -c '${command}'`
}
```

```typescript
// src/sandbox/manager.ts
export class SandboxManager {
  private config: SandboxConfig
  private platform: Platform
  
  async initialize(config: SandboxConfig): Promise<void> {
    this.config = config
    this.platform = detectPlatform()
    
    const deps = checkDependencies(this.platform)
    if (deps.errors.length > 0) {
      throw new Error(`Sandbox dependencies missing: ${deps.errors.join(', ')}`)
    }
    
    // Linux: 启动代理进程
    if (this.platform === 'linux' && config.network.httpProxyPort) {
      await this.startProxyProcess(config.network)
    }
  }
  
  async wrapCommand(command: string, shell: string = '/bin/bash'): Promise<string> {
    switch (this.platform) {
      case 'macos':
        return wrapMacOSCommand(command, this.config, shell)
      case 'linux':
      case 'wsl':
        return wrapLinuxCommand(
          command, 
          this.config, 
          shell,
          this.config.network.httpProxyPort
        )
      default:
        return command  // 无沙箱支持，原样返回
    }
  }
  
  cleanup(): void {
    // 清理临时文件
    // Linux: 停止代理进程
  }
}
```

```typescript
// src/sandbox/violations.ts
export class SandboxViolationMonitor {
  private violations: SandboxViolationEvent[] = []
  
  async start(): Promise<void> {
    // macOS: 监听系统日志
    if (platform === 'macos') {
      const logProcess = spawn('log', [
        'stream',
        '--predicate', 'process == "sandbox-exec"',
        '--style', 'compact'
      ])
      
      logProcess.stdout.on('data', (data) => {
        const line = data.toString()
        if (line.includes('deny')) {
          this.parseViolation(line)
        }
      })
    }
  }
  
  private parseViolation(logLine: string): void {
    // 解析: "Sandbox: npm(12345) deny file-read-data /etc/passwd"
    const match = logLine.match(/Sandbox: (\w+)\((\d+)\) deny (.+)/)
    if (match) {
      this.violations.push({
        command: match[1],
        timestamp: new Date(),
        line: match[3],
      })
    }
  }
  
  getViolations(): SandboxViolationEvent[] {
    return this.violations
  }
}
```
