本教程深入分析 Claude Code CLI 的输入处理系统与自动补全机制。

## 1. 输入处理概述

### 1.1 核心架构

Claude Code 的输入处理系统采用多层架构：

```
用户输入 → useInput (Ink) → PromptInput → TextInput/VimTextInput
                ↓
          输入过滤与转换
                ↓
          补全建议生成 (useTypeahead)
                ↓
          渲染建议列表 + Ghost Text
                ↓
          用户选择 → 应用补全 → 更新输入
```

**关键文件路径：**

| 功能 | 文件路径 |
|------|---------|
| 主输入组件 | `components/PromptInput/PromptInput.tsx` |
| 基础输入 | `components/TextInput.js` |
| Vim输入 | `components/VimTextInput.js` |
| 补全Hook | `hooks/useTypeahead.tsx` |
| 命令补全 | `utils/suggestions/commandSuggestions.ts` |
| 路径补全 | `utils/suggestions/directoryCompletion.ts` |
| Shell历史补全 | `utils/suggestions/shellHistoryCompletion.ts` |
| 历史记录管理 | `history.ts` |

```typescript
async function addToPromptHistory(command: HistoryEntry | string): Promise<void> {
  const entry = typeof command === 'string'
    ? { display: command, pastedContents: {} }
    : command
  
  // 处理粘贴内容存储
  const storedPastedContents: Record<number, StoredPastedContent> = {}
  for (const [id, content] of Object.entries(entry.pastedContents || {})) {

        type: content.type,
        contentHash: hash,

      void storePastedText(hash, content.content)
```

```typescript
// 核心刷新逻辑
async function immediateFlushHistory(): Promise<void> {
  if (pendingEntries.length === 0) return
  
  const historyPath = join(getClaudeConfigHomeDir(), 'history.jsonl')
  
  // 确保文件存在
  await writeFile(historyPath, '', { flag: 'a' })
  
  // 获取文件锁
  const release = await lock(historyPath, {
    stale: 10000,
    retries: { retries: 3, minTimeout: 50 }
  })
  
  // 写入JSONL行
  const jsonLines = pendingEntries.map(entry => jsonStringify(entry) + '\n')
  pendingEntries = []
  
  await appendFile(historyPath, jsonLines.join(''), { mode: 0o600 })
  
  // 释放锁
  await release()
}

// 带重试的刷新
async function flushPromptHistory(retries: number): Promise<void> {
  if (isWriting || pendingEntries.length === 0) return
  
  if (retries > 5) return  // 最多重试5次
  
  isWriting = true
  try {
    await immediateFlushHistory()
  } finally {
    isWriting = false
    if (pendingEntries.length > 0) {
      await sleep(500)
      void flushPromptHistory(retries + 1)
    }
  }
}
```

```typescript
// 获取历史（当前项目优先）
export async function* getHistory(): AsyncGenerator<HistoryEntry> {
  const currentProject = getProjectRoot()
  const currentSession = getSessionId()
  const otherSessionEntries: LogEntry[] = []
  let yielded = 0
  
  for await (const entry of makeLogEntryReader()) {
    if (!entry || typeof entry.project !== 'string') continue
    if (entry.project !== currentProject) continue
    
    // 当前会话优先
    if (entry.sessionId === currentSession) {
      yield await logEntryToHistoryEntry(entry)
      yielded++
    } else {
      otherSessionEntries.push(entry)
    }
    
    if (yielded + otherSessionEntries.length >= MAX_HISTORY_ITEMS) break
  }
  
  // 其他会话条目
  for (const entry of otherSessionEntries) {
    if (yielded >= MAX_HISTORY_ITEMS) return
    yield await logEntryToHistoryEntry(entry)
    yielded++
  }
}
```

```typescript
// 格式化粘贴文本引用
export function formatPastedTextRef(id: number, numLines: number): string {
  if (numLines === 0) {
    return `[Pasted text #${id}]`
  }
  return `[Pasted text #${id} +${numLines} lines]`
}

// 格式化图片引用
export function formatImageRef(id: number): string {
  return `[Image #${id}]`
}

// 解析引用
export function parseReferences(input: string): Array<{ id: number; match: string; index: number }> {
  const referencePattern = /\[(Pasted text|Image|\.\.\.Truncated text) #(\d+)(?: \+\d+ lines)?(\.)*\]/g
  const matches = [...input.matchAll(referencePattern)]
  return matches
    .map(match => ({
      id: parseInt(match[2] || '0'),
      match: match[0],
      index: match.index,
    }))
    .filter(match => match.id > 0)
}

// 展开粘贴文本引用
export function expandPastedTextRefs(
  input: string,
  pastedContents: Record<number, PastedContent>,
): string {
  const refs = parseReferences(input)
  let expanded = input
  
  // 从后向前替换以保持偏移正确
  for (let i = refs.length - 1; i >= 0; i--) {
    const ref = refs[i]!
    const content = pastedContents[ref.id]
    if (content?.type !== 'text') continue
    expanded = expanded.slice(0, ref.index) +
               content.content +
               expanded.slice(ref.index + ref.match.length)
  }
  return expanded
}
```

### 4.3 Token提取

```typescript
// Unicode感知的字符类
const AT_TOKEN_HEAD_RE = /^@[\p{L}\p{N}\p{M}_\-./\\()[\]~:]*/u
const PATH_CHAR_HEAD_RE = /^[\p{L}\p{N}\p{M}_\-./\\()[\]~:]+/u
const TOKEN_WITH_AT_RE = /(@[\p{L}\p{N}\p{M}_\-./\\()[\]~:]*|[\p{L}\p{N}\p{M}_\-./\\()[\]~:]+)$/u

// 提取可补全Token
export function extractCompletionToken(
  text: string,
  cursorPos: number,
  includeAtSymbol = false
): { token: string; startPos: number; isQuoted?: boolean } | null {
  if (!text) return null
  
  const textBeforeCursor = text.substring(0, cursorPos)
  
  // 检查引号@提及（如 @"my file with spaces")
  if (includeAtSymbol) {
    const quotedAtRegex = /@"([^"]*)"?$/
    const quotedMatch = textBeforeCursor.match(quotedAtRegex)
    if (quotedMatch && quotedMatch.index !== undefined) {
      const textAfterCursor = text.substring(cursorPos)
      const afterQuotedMatch = textAfterCursor.match(/^[^"]*"?/)
      const quotedSuffix = afterQuotedMatch ? afterQuotedMatch[0] : ''
      return {
        token: quotedMatch[0] + quotedSuffix,
        startPos: quotedMatch.index,
        isQuoted: true
      }
    }
  }
  
  // @ Token快速路径：使用lastIndexOf避免正则扫描
  if (includeAtSymbol) {
    const atIdx = textBeforeCursor.lastIndexOf('@')
    if (atIdx >= 0 && (atIdx === 0 || /\s/.test(textBeforeCursor[atIdx - 1]))) {
      const fromAt = textBeforeCursor.substring(atIdx)
      const atHeadMatch = fromAt.match(AT_TOKEN_HEAD_RE)
      if (atHeadMatch && atHeadMatch[0].length === fromAt.length) {
        const textAfterCursor = text.substring(cursorPos)
        const afterMatch = textAfterCursor.match(PATH_CHAR_HEAD_RE)
        const tokenSuffix = afterMatch ? afterMatch[0] : ''
        return {
          token: atHeadMatch[0] + tokenSuffix,
          startPos: atIdx,
          isQuoted: false
        }
      }
    }
  }
  
  // 非@Token：使用$锚定正则
  const tokenRegex = includeAtSymbol ? TOKEN_WITH_AT_RE : TOKEN_WITHOUT_AT_RE
  const match = textBeforeCursor.match(tokenRegex)
  if (!match || match.index === undefined) return null
  
  // 光标在Token中间时扩展到Token末尾
  const textAfterCursor = text.substring(cursorPos)
  const afterMatch = textAfterCursor.match(PATH_CHAR_HEAD_RE)
  const tokenSuffix = afterMatch ? afterMatch[0] : ''
  
  return {
    token: match[0] + tokenSuffix,
    startPos: match.index,
    isQuoted: false
  }
}
```

```typescript
// 应用路径补全
export function applyDirectorySuggestion(
  input: string,
  suggestionId: string,
  tokenStartPos: number,
  tokenLength: number,
  isDirectory: boolean
): { newInput: string; cursorPos: number } {
  const suffix = isDirectory ? '/' : ' '
  const before = input.slice(0, tokenStartPos)
  const after = input.slice(tokenStartPos + tokenLength)
  
  const replacement = '@' + suggestionId + suffix
  const newInput = before + replacement + after
  
  return {
    newInput,
    cursorPos: before.length + replacement.length
  }
}

// 应用Shell补全
export function applyShellSuggestion(
  suggestion: SuggestionItem,
  input: string,
  cursorOffset: number,
  onInputChange: (value: string) => void,
  setCursorOffset: (offset: number) => void,
  completionType: ShellCompletionType | undefined
): void {
  const beforeCursor = input.slice(0, cursorOffset)
  const lastSpaceIndex = beforeCursor.lastIndexOf(' ')
  const wordStart = lastSpaceIndex + 1
  
  let replacementText: string
  if (completionType === 'variable') {
    replacementText = '$' + suggestion.displayText + ' '
  } else if (completionType === 'command') {
    replacementText = suggestion.displayText + ' '
  } else {
    replacementText = suggestion.displayText
  }
  
  const newInput = input.slice(0, wordStart) + replacementText + input.slice(cursorOffset)
  onInputChange(newInput)
  setCursorOffset(wordStart + replacementText.length)
}
```

```typescript
import Fuse from 'fuse.js'

type CommandSearchItem = {
  descriptionKey: string[]
  partKey: string[] | undefined
  commandName: string
  command: Command
  aliasKey: string[] | undefined
}

// Fuse索引缓存
let fuseCache: {
  commands: Command[]
  fuse: Fuse<CommandSearchItem>
} | null = null

function getCommandFuse(commands: Command[]): Fuse<CommandSearchItem> {
  if (fuseCache?.commands === commands) {
    return fuseCache.fuse
  }
  
  const commandData: CommandSearchItem[] = commands
    .filter(cmd => !cmd.isHidden)
    .map(cmd => {
      const commandName = getCommandName(cmd)
      const parts = commandName.split(SEPARATORS).filter(Boolean)
      
      return {
        descriptionKey: (cmd.description ?? '')
          .split(' ')
          .map(word => cleanWord(word))
          .filter(Boolean),
        partKey: parts.length > 1 ? parts : undefined,
        commandName,
        command: cmd,
        aliasKey: cmd.aliases,
      }
    })
  
  const fuse = new Fuse(commandData, {
    includeScore: true,
    threshold: 0.3,    // 相对严格匹配
    location: 0,       // 优先匹配开头
    distance: 100,     // 允许描述匹配
```

```typescript
export function findMidInputSlashCommand(
  input: string,
  cursorOffset: number,
): MidInputSlashCommand | null {
  // 输入以 "/" 开头时由其他逻辑处理
  if (input.startsWith('/')) return null
  
  // 从光标向后查找空格后的 "/"
  const beforeCursor = input.slice(0, cursorOffset)
  const match = beforeCursor.match(/\s\/([a-zA-Z0-9_:-]*)$/)
  if (!match || match.index === undefined) return null
  
  const slashPos = match.index + 1
  const textAfterSlash = input.slice(slashPos + 1)
  const commandMatch = textAfterSlash.match(/^[a-zA-Z0-9_:-]*/)
  const fullCommand = commandMatch ? commandMatch[0] : ''
  
  // 光标在命令后（空格后）时不显示ghost text
  if (cursorOffset > slashPos + 1 + fullCommand.length) return null
  
  return {
    token: '/' + fullCommand,
    startPos: slashPos,
    partialCommand: fullCommand,
  }
}
```

```typescript
    for (const line of lines.reverse()) {  // 反向读取
      try {
        yield JSON.parse(line)
      } catch {}
    }
  }
}
```

### 10.2 最简命令补全

```typescript
// command-suggestions-simple.ts
import Fuse from 'fuse.js'

type Command = {
  name: string
  description?: string
  aliases?: string[]
}

export function generateCommandSuggestions(
  input: string,
  commands: Command[],
): Command[] {
  if (!input.startsWith('/')) return []
  
  const query = input.slice(1).toLowerCase()
  if (!query) return commands  // 显示所有
  
  const fuse = new Fuse(commands, {
    keys: [
      { name: 'name', weight: 3 },
      { name: 'aliases', weight: 2 },
      { name: 'description', weight: 0.5 },
    ],
    threshold: 0.3,
  })
  
  return fuse.search(query).map(r => r.item)
}
```

```typescript
// path-suggestions-simple.ts
import { readdir } from 'fs/promises'
import { join, basename, dirname } from 'path'

export async function getPathCompletions(
  partialPath: string,
  cwd: string = process.cwd(),
): Promise<string[]> {
  const dir = partialPath.includes('/')
    ? dirname(partialPath)
    : cwd
  
  const prefix = basename(partialPath)
  
  try {
    const entries = await readdir(dir, { withFileTypes: true })
    return entries
      .filter(e => e.name.startsWith(prefix))
      .map(e => e.isDirectory() ? e.name + '/' : e.name)
      .slice(0, 10)
  } catch {
    return []
  }
}
```

### 10.4 最简输入组件

```typescript
// InputComponent.tsx
import React, { useState, useCallback } from 'react'
import { Box, Text, useInput } from 'ink'

type Props = {
  onSubmit: (input: string) => void
  commands: Command[]
}

export function InputComponent({ onSubmit, commands }: Props): React.ReactNode {
  const [input, setInput] = useState('')
  const [suggestions, setSuggestions] = useState<Command[]>([])
  const [selectedSuggestion, setSelectedSuggestion] = useState(0)
  
  // 输入处理
  useInput((char, key) => {
    if (key.return) {
      onSubmit(input)
      setInput('')
      setSuggestions([])
      return
    }
    
    if (key.backspace) {
      setInput(prev => prev.slice(0, -1))
      return
    }
    
    if (key.tab && suggestions.length > 0) {
      const cmd = suggestions[selectedSuggestion]
      setInput('/' + cmd.name + ' ')
      setSuggestions([])
      return
    }
    
    if (key.upArrow && suggestions.length > 0) {
      setSelectedSuggestion(prev => Math.max(0, prev - 1))
      return
    }
    
    if (key.downArrow && suggestions.length > 0) {
      setSelectedSuggestion(prev => Math.min(suggestions.length - 1, prev + 1))
      return
    }
    
    // 添加字符
    setInput(prev => prev + char)
    
    // 生成建议
    if (char === '/') {
      setSuggestions(commands)
    } else if (input.startsWith('/')) {
      setSuggestions(generateCommandSuggestions(input + char, commands))
    } else {
      setSuggestions([])
    }
  })
  
  return (
    <Box flexDirection="column">
      <Box>
        <Text>Input: {input}</Text>
      </Box>
      {suggestions.length > 0 && (
        <Box flexDirection="column">
          {suggestions.map((cmd, i) => (
            <Text
              key={cmd.name}
              color={i === selectedSuggestion ? 'cyan' : 'gray'}
            >
              {i === selectedSuggestion ? '> ' : '  '}/{cmd.name}
            </Text>
          ))}
        </Box>
      )}
    </Box>
  )
}
```

```typescript
// useTypeahead-simple.ts
import { useState, useCallback, useMemo } from 'react'

export function useTypeahead(
  input: string,
  cursorOffset: number,
  commands: Command[],
): {
  suggestions: SuggestionItem[]
  selectedSuggestion: number
  handleKeyDown: (key: Key) => void
} {
  const [suggestions, setSuggestions] = useState<SuggestionItem[]>([])
  const [selectedSuggestion, setSelectedSuggestion] = useState(0)
  
  // 输入变更时生成建议
  useMemo(() => {
    if (input.startsWith('/')) {
      const cmds = generateCommandSuggestions(input, commands)
      setSuggestions(cmds.map(cmd => ({
        id: cmd.name,
        displayText: '/' + cmd.name,
        description: cmd.description,
        metadata: cmd,
      })))
      setSelectedSuggestion(0)
    } else {
      setSuggestions([])
    }
  }, [input, commands])
  
  // 键盘处理
  const handleKeyDown = useCallback((key: Key) => {
    if (key.upArrow && suggestions.length > 0) {
      setSelectedSuggestion(prev => Math.max(0, prev - 1))
    }
    if (key.downArrow && suggestions.length > 0) {
      setSelectedSuggestion(prev => Math.min(suggestions.length - 1, prev + 1))
    }
  }, [suggestions.length])
  
  return { suggestions, selectedSuggestion, handleKeyDown }
}
```

## 12. 常见问题解答

**Q: 为什么历史有时显示不同会话的命令？**
A: `getHistory`优先当前会话，然后是其他会话，避免并发会话交错。

**Q: Tab补全如何区分命令和路径？**
A: 根据输入前缀：`/`触发命令补全，`@`触发路径补全。

**Q: 如何处理带空格的路径？**
A: 使用`@"path with spaces"`格式，extractCompletionToken识别引号模式。

**Q: 为什么Fuse索引要缓存？**
A: commands数组稳定（REPL中memoized），缓存避免每次按键重建索引。

**Q: 历史搜索的光标定位逻辑？**
A: 定位到匹配位置，使用getValueFromInput计算相对于clean值的位置。
