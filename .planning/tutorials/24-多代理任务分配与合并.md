# 多代理任务分配与结果合并机制

**分析日期:** 2026-04-02

## 1. 概述

### 多代理架构设计目标

Claude Code CLI的多代理系统旨在实现：

1. **并行任务执行** - 同时启动多个独立代理处理不同任务，最大化吞吐量
2. **任务隔离** - 每个代理拥有独立的上下文、工具权限和生命周期
3. **统一协调** - Coordinator模式提供集中式的任务分发和结果汇总
4. **异步执行** - 后台代理不阻塞主线程，通过通知机制返回结果

### 并行任务执行的优势

```
传统串行执行:
Task A (30s) → Task B (45s) → Task C (20s) = 95秒总时长

多代理并行执行:
Task A (30s) ┐
Task B (45s) ├→ 同时执行 = 45秒总时长（最慢任务决定）
Task C (20s) ┘
```

**核心收益:**
- 研究任务可完全并行（只读操作无冲突）
- 实现任务按文件集分组并行
- 验证任务可与实现任务并行执行（不同文件区域）

```typescript
    selectedAgent.background === true || 
    isCoordinator || 
    forceAsync;

    }
```

### 任务调度算法

```typescript
// messageQueueManager.ts (行151-193) - 优先级队列调度

const PRIORITY_ORDER: Record<QueuePriority, number> = {
  now: 0,    // 最高优先级（立即处理）
  next: 1,   // 用户输入默认优先级
  later: 2   // 任务通知默认优先级
};

export function dequeue(filter?: (cmd) => boolean): QueuedCommand | undefined {
  // 查找最高优先级命令（尊重filter）
  let bestIdx = -1;
  let bestPriority = Infinity;
  
  for (let i = 0; i < commandQueue.length; i++) {
    const cmd = commandQueue[i];
    if (filter && !filter(cmd)) continue;  // 过滤不匹配命令
    
    const priority = PRIORITY_ORDER[cmd.priority ?? 'next'];
    if (priority < bestPriority) {
      bestIdx = i;
      bestPriority = priority;
    }
  }
  
  // 移除并返回最高优先级命令
  return commandQueue.splice(bestIdx, 1)[0];
}
```

```typescript
      }
      return false;
    }
    
    return true;
  });
}
```

```typescript
// agentToolUtils.ts runAsyncAgentLifecycle (行508-686)

export async function runAsyncAgentLifecycle({
  taskId,
  abortController,
  makeStream,  // runAgent的生成器
  metadata,
  ...
}): Promise<void> {
  const agentMessages: MessageType[] = [];
  const tracker = createProgressTracker();  // 进度追踪器
  
  try {
    // 1. 流式收集消息
    for await (const message of makeStream(onCacheSafeParams)) {
      agentMessages.push(message);
      
      // 2. 实时更新UI（如果retain）
      rootSetAppState(prev => {
        const t = prev.tasks[taskId];
        if (!isLocalAgentTask(t) || !t.retain) return prev;
        return {
          ...prev,
          tasks: {
            ...prev.tasks,
            [taskId]: { ...t, messages: [...t.messages ?? [], message] }
          }
        };
      });
      
      // 3. 更新进度追踪
      updateProgressFromMessage(tracker, message, resolveActivity, tools);
      updateAsyncAgentProgress(taskId, getProgressUpdate(tracker), rootSetAppState);
      
      // 4. 发送SDK进度事件
      const lastToolName = getLastToolUseName(message);
      if (lastToolName) {
        emitTaskProgress(tracker, taskId, toolUseId, description, startTime, lastToolName);
      }
    }
    
    // 5. 最终化结果
    const agentResult = finalizeAgentTool(agentMessages, taskId, metadata);
    completeAsyncAgent(agentResult, rootSetAppState);
    
    // 6. 发送完成通知
    enqueueAgentNotification({
      taskId,
      description,
      status: 'completed',
      setAppState: rootSetAppState,
      finalMessage: extractTextContent(agentResult.content, '\n'),
      usage: {
        totalTokens: getTokenCountFromTracker(tracker),
        toolUses: agentResult.totalToolUseCount,
        durationMs: agentResult.totalDurationMs
      }
    });
  } catch (error) {
    // 7. 错误处理
    if (error instanceof AbortError) {
      killAsyncAgent(taskId, rootSetAppState);
      enqueueAgentNotification({ taskId, status: 'killed', ... });
    } else {
      failAsyncAgent(taskId, errorMessage(error), rootSetAppState);
      enqueueAgentNotification({ taskId, status: 'failed', error: msg, ... });
    }
  }
}
```

```typescript
// LocalAgentTask.tsx 进度追踪 (行33-104)

export type ProgressTracker = {
  toolUseCount: number;
  latestInputTokens: number;      // 输入token（累计）
  cumulativeOutputTokens: number; // 输出token（累加）
  recentActivities: ToolActivity[];  // 最近5个工具活动
};

export function updateProgressFromMessage(
  tracker: ProgressTracker,
  message: Message,
  resolveActivityDescription?: ActivityDescriptionResolver
): void {
  if (message.type !== 'assistant') return;
  
  const usage = message.message.usage;
  
  // 更新token计数
  tracker.latestInputTokens = usage.input_tokens + 
    (usage.cache_creation_input_tokens ?? 0) + 
    (usage.cache_read_input_tokens ?? 0);
  tracker.cumulativeOutputTokens += usage.output_tokens;
  
  // 收集工具活动
  for (const content of message.message.content) {
    if (content.type === 'tool_use') {
      tracker.toolUseCount++;
      tracker.recentActivities.push({
        toolName: content.name,
        input: content.input,
        activityDescription: resolveActivityDescription?.(content.name, content.input),
        isSearch: classification?.isSearch,
        isRead: classification?.isRead
      });
    }
  }
  
  // 保留最近5个活动
  while (tracker.recentActivities.length > MAX_RECENT_ACTIVITIES) {
    tracker.recentActivities.shift();
  }
}
```

```typescript
// 核心调度器架构

class AgentScheduler {
  private taskQueue: PriorityQueue<Task>;
  private runningTasks: Map<string, AgentTask>;
  private notifyQueue: Queue<Notification>;
  
  // 任务提交
  async submitTask(task: TaskSpec): Promise<TaskHandle> {
    const taskId = generateTaskId();
    const abortController = new AbortController();
    
    // 注册任务
    const handle = {
      taskId,
      abortController,
      status: 'pending',
      outputFile: getTaskOutputPath(taskId)
    };
    
    this.taskQueue.enqueue(task, task.priority);
    
    // 启动执行（异步）
    void this.executeTask(taskId, task, abortController.signal);
    
    return handle;
  }
  
  // 任务执行
  private async executeTask(
    taskId: string,
    task: TaskSpec,
    signal: AbortSignal
  ): Promise<void> {
    try {
      // 更新状态为running
      this.updateStatus(taskId, 'running');
      
      // 创建代理上下文
      const agentContext = createSubagentContext({
        tools: resolveAgentTools(task.agentDef),
        permissionMode: task.permissionMode,
        abortSignal: signal
      });
      
      // 流式执行
      const messages: Message[] = [];
      for await (const msg of runAgentLoop(agentContext, task.prompt)) {
        messages.push(msg);
        
        // 实时进度更新
        this.emitProgress(taskId, msg);
        
        // 检查中止
        if (signal.aborted) throw new AbortError();
      }
      
      // 完成
      const result = finalizeResult(messages);
      this.updateStatus(taskId, 'completed');
      this.enqueueNotification(taskId, 'completed', result);
      
    } catch (error) {
      if (error instanceof AbortError) {
        this.updateStatus(taskId, 'killed');
        this.enqueueNotification(taskId, 'killed');
      } else {
        this.updateStatus(taskId, 'failed');
        this.enqueueNotification(taskId, 'failed', { error: error.message });
      }
    }
  }
  
  // 中止任务
  async killTask(taskId: string): Promise<void> {
    const task = this.runningTasks.get(taskId);
    if (task && task.status === 'running') {
      task.abortController.abort();
    }
  }
  
  // 通知入队
  private enqueueNotification(
    taskId: string,
    status: 'completed' | 'failed' | 'killed',
    result?: any
  ): void {
    const notification = formatNotification(taskId, status, result);
    this.notifyQueue.enqueue(notification, 'later');  // 低于用户输入优先级
  }
}
```

### 流式合并算法

```typescript
// 多代理结果流式合并器

class StreamMerger {
  private streams: Map<string, AsyncGenerator<Message>>;
  private mergedStream: AsyncGenerator<Message>;
  
  // 注册代理流
  registerStream(agentId: string, stream: AsyncGenerator<Message>): void {
    this.streams.set(agentId, stream);
  }
  
  // 合并所有流（并发迭代）
  async *merge(): AsyncGenerator<MergedEvent> {
    const iterators = Array.from(this.streams.entries())
      .map(([id, stream]) => ({
        agentId: id,
        iterator: stream[Symbol.asyncIterator]()
      }));
    
    // 并发拉取
    const pending = new Map<string, Promise<IteratorResult<Message>>>();
    for (const { agentId, iterator } of iterators) {
      pending.set(agentId, iterator.next());
    }
    
    while (pending.size > 0) {
      // 等待任意流产出
      const settled = await Promise.race(
        Array.from(pending.entries())
          .map(([id, prom]) => prom.then(r => ({ agentId: id, result: r })))
      );
      
      if (!settled.result.done) {
        // 产出消息
        yield {
          type: 'message',
          agentId: settled.agentId,
          message: settled.result.value
        };
        
        // 重新拉取该流
        const iterator = iterators.find(i => i.agentId === settled.agentId)!.iterator;
        pending.set(settled.agentId, iterator.next());
      } else {
        // 流结束
        pending.delete(settled.agentId);
        
        yield {
          type: 'stream_end',
          agentId: settled.agentId
        };
      }
    }
  }
}

// 使用示例:
const merger = new StreamMerger();
merger.registerStream('agent-1', runAgent(spec1));
merger.registerStream('agent-2', runAgent(spec2));

for await (const event of merger.merge()) {
  if (event.type === 'message') {
    console.log(`[${event.agentId}] ${event.message}`);
  } else if (event.type === 'stream_end') {
    console.log(`[${event.agentId}] completed`);
  }
}
```

### 优先级队列实现

```typescript
// 三级优先级队列

type Priority = 'now' | 'next' | 'later';

interface QueuedCommand {
  value: string | ContentBlockParam[];
  priority: Priority;
  mode: PromptInputMode;
  toolUseId?: string;
}

class PriorityCommandQueue {
  private queues: {
    now: QueuedCommand[];
    next: QueuedCommand[];
    later: QueuedCommand[];
  } = { now: [], next: [], later: [] };
  
  // 入队
  enqueue(cmd: QueuedCommand): void {
    const priority = cmd.priority ?? 'next';
    this.queues[priority].push(cmd);
    this.notifySubscribers();
  }
  
  // 出队（最高优先级优先）
  dequeue(filter?: (cmd: QueuedCommand) => boolean): QueuedCommand | undefined {
    const priorities: Priority[] = ['now', 'next', 'later'];
    
    for (const p of priorities) {
      const queue = this.queues[p];
      
      if (filter) {
        // 查找第一个匹配的命令
        const idx = queue.findIndex(filter);
        if (idx !== -1) {
          const [cmd] = queue.splice(idx, 1);
          this.notifySubscribers();
          return cmd;
        }
      } else if (queue.length > 0) {
        const cmd = queue.shift()!;
        this.notifySubscribers();
        return cmd;
      }
    }
    
    return undefined;
  }
  
  // 订阅机制（React useSyncExternalStore兼容）
  private subscribers: Set<() => void> = new Set();
  
  subscribe(callback: () => void): () => void {
    this.subscribers.add(callback);
    return () => this.subscribers.delete(callback);
  }
  
  private notifySubscribers(): void {
    this.subscribers.forEach(cb => cb());
  }
  
  getSnapshot(): readonly QueuedCommand[] {
    return [
      ...this.queues.now,
      ...this.queues.next,
      ...this.queues.later
    ];
  }
}
```

### 工具权限过滤器

```typescript
// 代理工具权限控制

class ToolPermissionFilter {
  private globalDisallowed: Set<string>;
  private asyncAllowed: Set<string>;
  
  constructor() {
    this.globalDisallowed = new Set([
      'TaskOutput',        // 防止递归
      'ExitPlanMode',      // 主线程抽象
      'EnterPlanMode',
      'TaskStop',          // 需要主线程状态
      'AskUserQuestion',   // 无法显示UI
      'Agent'              // 防止嵌套（可配置）
    ]);
    
    this.asyncAllowed = new Set([
      'Read', 'Edit', 'Write',
      'Bash', 'Grep', 'Glob',
      'WebSearch', 'WebFetch',
      'TodoWrite',
      'Skill',
      'NotebookEdit'
    ]);
  }
  
  // 过滤工具池
  filterTools(
    tools: Tool[],
    options: {
      isBuiltIn: boolean;
      isAsync: boolean;
      permissionMode?: string;
    }
  ): Tool[] {
    return tools.filter(tool => {
      // MCP工具始终允许
      if (tool.name.startsWith('mcp__')) return true;
      
      // 全局禁止检查
      if (this.globalDisallowed.has(tool.name)) return false;
      
      // 异步代理限制
      if (options.isAsync && !this.asyncAllowed.has(tool.name)) {
        return false;
      }
      
      return true;
    });
  }
  
  // 解析代理工具列表
  resolveAgentTools(
    agentDef: AgentDefinition,
    availableTools: Tool[]
  ): ResolvedTools {
    // 通配符 '*' → 所有允许工具
    if (!agentDef.tools || agentDef.tools.includes('*')) {
      return {
        hasWildcard: true,
        resolvedTools: this.filterTools(availableTools, {
          isBuiltIn: agentDef.source === 'built-in',
          isAsync: true
        })
      };
    }
    
    // 显式工具列表
    const resolved: Tool[] = [];
    for (const toolSpec of agentDef.tools) {
      const toolName = parseToolName(toolSpec);
      const tool = availableTools.find(t => t.name === toolName);
      
      if (tool) {
        resolved.push(tool);
      }
    }
    
    return { hasWildcard: false, resolvedTools: resolved };
  }
}
```
