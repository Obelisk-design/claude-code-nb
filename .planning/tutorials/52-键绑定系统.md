# 52-键绑定系统详解

## 1. 键绑定概述

Claude Code CLI 的键绑定系统是一个高度可配置的键盘快捷键管理框架，支持：
- **多上下文绑定**：不同UI场景使用不同的快捷键
- **用户自定义**：用户可通过配置文件覆盖默认绑定
- **和弦序列**：支持多键组合序列（如 `ctrl+k ctrl+s`）
- **平台适配**：自动适配 Windows/macOS/Linux 的按键差异
- **实时重载**：修改配置文件后自动生效

### 核心架构

```
keybindings/
├── defaultBindings.ts      # 默认绑定配置
├── schema.ts               # Zod schema定义和验证
├── parser.ts               # 按键字符串解析器
├── resolver.ts             # 按键匹配和解析器
├── match.ts                # Ink按键匹配逻辑
├── validate.ts             # 配置验证和错误检查
├── reservedShortcuts.ts    # 系统保留快捷键
├── loadUserBindings.ts     # 用户配置加载和热重载
├── KeybindingContext.tsx   # React Context提供器
├── KeybindingProviderSetup.tsx  # 完整Provider设置
├── useKeybinding.ts        # React Hook for绑定处理
├── useShortcutDisplay.ts   # 显示文本Hook
├── shortcutFormat.ts       # 非React环境的显示文本
└── template.ts             # 配置文件模板生成
```

```typescript
export function parseKeystroke(input: string): ParsedKeystroke {
  const parts = input.split('+')           // 拆分修饰键
  const keystroke: ParsedKeystroke = {
    key: '', ctrl: false, alt: false,
    shift: false, meta: false, super: false,
  }
  
  for (const part of parts) {
    const lower = part.toLowerCase()
    switch (lower) {
      case 'ctrl': case 'control':
        keystroke.ctrl = true
        break
      case 'alt': case 'opt': case 'option':
        keystroke.alt = true               // Alt别名
        break
      case 'shift':
        keystroke.shift = true
        break
      case 'meta':
        keystroke.meta = true
        break
      case 'cmd': case 'command': case 'super': case 'win':
        keystroke.super = true             // Cmd别名
        break
      case 'esc':
        keystroke.key = 'escape'           // 键名规范化
        break
      case 'return':
        keystroke.key = 'enter'
        break
      case '↑': keystroke.key = 'up'       // Unicode箭头
        break
      case '↓': keystroke.key = 'down'
        break
      // ...其他特殊键
      default:
        keystroke.key = lower              // 普通键
        break
    }
  }
  return keystroke
}
```

```typescript
export function parseChord(input: string): Chord {
```

```typescript
export function resolveKey(
  input: string,                      // Ink输入字符
  key: Key,                           // Ink Key对象（含修饰键）
  activeContexts: KeybindingContextName[],  // 当前激活的上下文
  bindings: ParsedBinding[],          // 所有解析后的绑定
): ResolveResult {
  let match: ParsedBinding | undefined
  const ctxSet = new Set(activeContexts)
  
  for (const binding of bindings) {
    // 只处理单键绑定
    if (binding.chord.length !== 1) continue
    if (!ctxSet.has(binding.context)) continue
    
    if (matchesBinding(input, key, binding)) {
      match = binding                   // 最后一个匹配胜出
    }
  }
  
  if (!match) return { type: 'none' }
  if (match.action === null) return { type: 'unbound' }
  return { type: 'match', action: match.action }
}
```

**和弦解析**（第166-244行）:

```typescript
export function resolveKeyWithChordState(
  input: string,
  key: Key,
  activeContexts: KeybindingContextName[],
  bindings: ParsedBinding[],
  pending: ParsedKeystroke[] | null,   // 当前pending和弦状态
): ChordResolveResult {
  // ESC取消和弦
  if (key.escape && pending !== null) {
    return { type: 'chord_cancelled' }
  }
  
  // 构建当前按键
  const currentKeystroke = buildKeystroke(input, key)
  if (!currentKeystroke) {
    if (pending !== null) return { type: 'chord_cancelled' }
    return { type: 'none' }
  }
  
  // 构建测试和弦
  const testChord = pending
    ? [...pending, currentKeystroke]
    : [currentKeystroke]
  
  // 检查是否可能是更长和弦的前缀
  const chordWinners = new Map<string, string | null>()
  for (const binding of contextBindings) {
    if (
      binding.chord.length > testChord.length &&
      chordPrefixMatches(testChord, binding)
    ) {
      chordWinners.set(chordToString(binding.chord), binding.action)
    }
  }
  
  let hasLongerChords = false
  for (const action of chordWinners.values()) {
    if (action !== null) {
      hasLongerChords = true
      break
    }
  }
  
  // 如果可能是更长和弦，优先等待
  if (hasLongerChords) {
    return { type: 'chord_started', pending: testChord }
  }
  
  // 检查精确匹配
  let exactMatch: ParsedBinding | undefined
  for (const binding of contextBindings) {
    if (chordExactlyMatches(testChord, binding)) {
      exactMatch = binding
    }
  }
  
  if (exactMatch) {
    if (exactMatch.action === null) return { type: 'unbound' }
    return { type: 'match', action: exactMatch.action }
  }
  
  // 无匹配且在和弦中 -> 取消
  if (pending !== null) return { type: 'chord_cancelled' }
  
  return { type: 'none' }
}
```

```typescript
export function keystrokesEqual(a: ParsedKeystroke, b: ParsedKeystroke): boolean {
  return (
    a.key === b.key &&
    a.ctrl === b.ctrl &&
    a.shift === b.shift &&
    // Alt和Meta在终端中不可区分，视为相同
    (a.alt || a.meta) === (b.alt || b.meta) &&
    a.super === b.super            // Super是独立修饰键
  )
}
```

**和弦匹配辅助函数**（第123-152行）:

```typescript
// 检查前缀匹配（用于检测潜在和弦）
function chordPrefixMatches(prefix: ParsedKeystroke[], binding: ParsedBinding): boolean {
  if (prefix.length >= binding.chord.length) return false
  for (let i = 0; i < prefix.length; i++) {
    if (!keystrokesEqual(prefix[i], binding.chord[i])) return false
  }
  return true
}

// 检查精确匹配
function chordExactlyMatches(chord: ParsedKeystroke[], binding: ParsedBinding): boolean {
  if (chord.length !== binding.chord.length) return false
  for (let i = 0; i < chord.length; i++) {
    if (!keystrokesEqual(chord[i], binding.chord[i])) return false
  }
  return true
}
```

**关键设计**:

1. **和弦优先**: 优先等待可能的和弦序列，避免触发单键
2. **Alt/Meta折叠**: 终端限制，无法区分alt和meta
3. **取消机制**: ESC或无效键取消和弦
4. **最后胜出**: 用户配置覆盖默认（数组顺序）

```typescript
export function matchesKeystroke(
  input: string,
  key: Key,
  target: ParsedKeystroke,
): boolean {
  const keyName = getKeyName(input, key)
  if (keyName !== target.key) return false
  
  const inkMods = getInkModifiers(key)
  
  // ESC特殊处理：Ink设置key.meta=true（历史原因）
  if (key.escape) {
    return modifiersMatch({ ...inkMods, meta: false }, target)
  }
  
  return modifiersMatch(inkMods, target)
}
```

2. **ESC特殊处理**: ESC按下时Ink错误设置meta=true
3. **终端限制**: Alt和Meta在终端中发送相同序列

```typescript
function validateKeystroke(keystroke: string): KeybindingWarning | null {
  const parts = keystroke.toLowerCase().split('+')
  
  for (const part of parts) {
    const trimmed = part.trim()
    if (!trimmed) {
      return {
        type: 'parse_error',
        severity: 'error',
        message: `Empty key part in "${keystroke}"`,
        key: keystroke,
        suggestion: 'Remove extra "+" characters',
      }
    }
  }
  
  // 尝试解析
  const parsed = parseKeystroke(keystroke)
  if (!parsed.key && !parsed.ctrl && !parsed.alt && !parsed.shift && !parsed.meta) {
    return {
      type: 'parse_error',
      severity: 'error',
      message: `Could not parse keystroke "${keystroke}"`,
      key: keystroke,
    }
  }
  
  return null
}
```

**JSON重复键检测**（第258-307行）:

```typescript
export function checkDuplicateKeysInJson(jsonString: string): KeybindingWarning[] {
  const warnings: KeybindingWarning[] = []
  
  // 匹配bindings块
  const bindingsBlockPattern = /"bindings"\s*:\s*\{([^{}]*(?:\{[^{}]*\}[^{}]*)*)\}/g
  
  let blockMatch
  while ((blockMatch = bindingsBlockPattern.exec(jsonString)) !== null) {
    const blockContent = blockMatch[1]
    if (!blockContent) continue
    
    // 查找上下文名称
    const textBeforeBlock = jsonString.slice(0, blockMatch.index)
    const contextMatch = textBeforeBlock.match(/"context"\s*:\s*"([^"]+)"[^{]*$/)
    const context = contextMatch?.[1] ?? 'unknown'
    
    // 统计键出现次数
    const keyPattern = /"([^"]+)"\s*:/g
    const keysByName = new Map<string, number>()
    
    let keyMatch
    while ((keyMatch = keyPattern.exec(blockContent)) !== null) {
      const key = keyMatch[1]
      if (!key) continue
      
      const count = (keysByName.get(key) ?? 0) + 1
      keysByName.set(key, count)
      
      if (count === 2) {
        warnings.push({
          type: 'duplicate',
          severity: 'warning',
          message: `Duplicate key "${key}" in ${context} bindings`,
          key,
```

```typescript
function normalizeStep(step: string): string {
  const parts = step.split('+')
  const modifiers: string[] = []
  let mainKey = ''
  
  for (const part of parts) {
    const lower = part.trim().toLowerCase()
    if (['ctrl', 'control', 'alt', 'opt', 'option', 'meta', 'cmd', 'command', 'shift'].includes(lower)) {
      // 规范化修饰键名称
      if (lower === 'control') modifiers.push('ctrl')
      else if (lower === 'option' || lower === 'opt') modifiers.push('alt')
      else if (lower === 'command' || lower === 'cmd') modifiers.push('cmd')
      else modifiers.push(lower)
    } else {
      mainKey = lower
    }
  }
  
  modifiers.sort()            // 排序确保一致性
  return [...modifiers, mainKey].join('+')
}
```

```typescript
// parser.ts

export function parseKeystroke(input: string): ParsedKeystroke {
  const parts = input.split('+')
  const keystroke: ParsedKeystroke = {
    key: '',
    ctrl: false,
    alt: false,
    shift: false,
    meta: false,
    super: false,
  }
  
  for (const part of parts) {
    const lower = part.toLowerCase()
    switch (lower) {
      case 'ctrl': keystroke.ctrl = true; break
      case 'alt': keystroke.alt = true; break
      case 'shift': keystroke.shift = true; break
      case 'meta': keystroke.meta = true; break
      case 'cmd': keystroke.super = true; break
      default: keystroke.key = lower; break
    }
  }
  
  return keystroke
}

export function parseChord(input: string): Chord {
  if (input === ' ') return [parseKeystroke('space')]
  return input.trim().split(/\s+/).map(parseKeystroke)
}

export function parseBindings(blocks: KeybindingBlock[]): ParsedBinding[] {
  const bindings: ParsedBinding[] = []
  for (const block of blocks) {
    for (const [key, action] of Object.entries(block.bindings)) {
      bindings.push({
        chord: parseChord(key),
        action,
        context: block.context,
      })
    }
  }
  return bindings
}
```

```typescript
import type { Key } from 'ink'

export function getKeyName(input: string, key: Key): string | null {
  if (key.escape) return 'escape'
  if (key.return) return 'enter'
  if (key.tab) return 'tab'
  if (key.upArrow) return 'up'
  if (key.downArrow) return 'down'
  if (input.length === 1) return input.toLowerCase()
  return null
}

export function matchesKeystroke(
  input: string,
  key: Key,
  target: ParsedKeystroke,
): boolean {
  const keyName = getKeyName(input, key)
  if (keyName !== target.key) return false
  
  if (key.ctrl !== target.ctrl) return false
  if (key.shift !== target.shift) return false
  if (key.meta !== (target.alt || target.meta)) return false
  if (key.super !== target.super) return false
  
  return true
}
```

```typescript
export function validateBindings(
  userBlocks: KeybindingBlock[],
): KeybindingWarning[] {
  const warnings: KeybindingWarning[] = []
  
  const validContexts = ['Global', 'Chat', 'Autocomplete', ...]
  
  for (const block of userBlocks) {
    // 检查上下文
    if (!validContexts.includes(block.context)) {
      warnings.push({
        type: 'invalid_context',
        severity: 'error',
        message: `Unknown context "${block.context}"`,
      })
    }
    
    // 检查按键格式
    for (const [key, action] of Object.entries(block.bindings)) {
      const parsed = parseKeystroke(key)
      if (!parsed.key && !parsed.ctrl && !parsed.alt) {
        warnings.push({
          type: 'parse_error',
          severity: 'error',
          message: `Could not parse "${key}"`,
        })
      }
    }
  }
  
  return warnings
}
```

```typescript
  // 同步ref和state
  const setPending = (pending: ParsedKeystroke[] | null) => {
    pendingChordRef.current = pending
    setPendingChord(pending)
  }
  
  const resolve = (input: string, key: Key, contexts: KeybindingContextName[]) => 
    resolveKeyWithChordState(input, key, contexts, bindings, pendingChordRef.current)
  
  return (
    <KeybindingContext.Provider value={{ bindings, resolve, pendingChord, setPendingChord: setPending }}>
      {children}
    </KeybindingContext.Provider>
  )
}

export function useKeybindingContext() {
  return useContext(KeybindingContext)!
}
```

### 7.9 步骤8: 实现Hook接口

```typescript
// useKeybinding.ts

import { useInput } from 'ink'
import { useKeybindingContext } from './KeybindingContext'

export function useKeybinding(
  action: string,
  handler: () => void,
  options: { context?: KeybindingContextName } = {},
) {
  const { context = 'Global' } = options
  const ctx = useKeybindingContext()
  
  useInput((input, key) => {
    const contexts = [context, 'Global']
    const result = ctx.resolve(input, key, contexts)
    
    if (result.type === 'match' && result.action === action) {
      handler()
    }
  })
}

export function useKeybindings(
  handlers: Record<string, () => void>,
  options: { context?: KeybindingContextName } = {},
) {
  const { context = 'Global' } = options
  const ctx = useKeybindingContext()
  
  useInput((input, key) => {
    const contexts = [context, 'Global']
    const result = ctx.resolve(input, key, contexts)
    
    if (result.type === 'match' && result.action in handlers) {
      handlers[result.action]()
    }
  })
}
```

## 8. 总结

Claude Code CLI 的键绑定系统是一个精心设计的框架：

**核心特点**:
1. **分层上下文**: 不同UI区域独立绑定
2. **平台适配**: 自动处理Windows/macOS/Linux差异
3. **和弦序列**: 支持多键组合序列
4. **用户自定义**: 灵活的配置覆盖机制
5. **实时验证**: 配置错误即时提示
6. **热重载**: 修改配置自动生效

**关键实现**:
- `parser.ts`: 按键字符串解析
- `resolver.ts`: 按键匹配和和弦处理
- `match.ts`: Ink按键对象适配
- `validate.ts`: 配置验证和警告生成
- `loadUserBindings.ts`: 配置加载和热重载
- `KeybindingContext.tsx`: React状态管理
- `useKeybinding.ts`: 组件接口Hook

**设计亮点**:
1. **纯函数解析器**: 无副作用，易于测试
2. **最后胜出规则**: 用户配置自然覆盖默认
3. **类型安全**: Zod schema验证配置
4. **优雅降级**: 配置错误时使用默认
5. **终端兼容**: 正确处理Alt/Meta混淆
