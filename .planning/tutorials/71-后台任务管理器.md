## 1. 后台任务概述

### 1.1 什么是后台任务

后台任务管理器是Claude Code CLI的核心组件，负责管理所有异步执行的长时间运行操作。它允许用户在等待命令执行的同时继续进行对话交互。

**核心功能:**
- Shell命令后台执行 (`LocalShellTask`)
- Agent任务后台运行 (`LocalAgentTask`)
- 远程Agent任务 (`RemoteAgentTask`)
- Teammate协作任务 (`InProcessTeammateTask`)
- 工作流任务 (`LocalWorkflowTask`)
- MCP监控任务 (`MonitorMcpTask`)
- 记忆整合任务 (`DreamTask`)

### 1.2 任务类型定义

**文件位置:** `Task.ts`

```typescript
export type TaskType =
  | 'local_bash'      // 本地Shell命令
  | 'local_agent'     // 本地Agent任务
  | 'remote_agent'    // 远程Agent任务
  | 'in_process_teammate'  // 进程内协作任务
  | 'local_workflow'  // 本地工作流
  | 'monitor_mcp'     // MCP监控
  | 'dream'           // 记忆整合
```

### 1.3 任务状态生命周期

```typescript
export type TaskStatus =
  | 'pending'     // 等待执行
  | 'running'     // 正在运行
  | 'completed'   // 成功完成
  | 'failed'      // 执行失败
  | 'killed'      // 被手动终止
```

**状态转换图:**
```
pending → running → completed/failed/killed
                ↑
            (可被终止)
```

### 3.2 任务创建流程

**文件位置:** `tasks/LocalShellTask/LocalShellTask.tsx`

```typescript
export async function spawnShellTask(
  input: LocalShellSpawnInput & { shellCommand: ShellCommand },
  context: TaskContext
): Promise<TaskHandle> {
  const { command, description, shellCommand, toolUseId, agentId, kind } = input
  const { setAppState } = context

  // 1. 获取任务ID（由TaskOutput管理）
  const taskId = shellCommand.taskOutput.taskId

  // 2. 注册进程退出清理
  const unregisterCleanup = registerCleanup(async () => {
    killTask(taskId, setAppState)
  })

  // 3. 创建任务状态
  const taskState: LocalShellTaskState = {
    ...createTaskStateBase(taskId, 'local_bash', description, toolUseId),
    type: 'local_bash',
    status: 'running',
    command,
    completionStatusSentInAttachment: false,
    shellCommand,
    unregisterCleanup,
    lastReportedTotalLines: 0,
    isBackgrounded: true,  // 直接后台化
    agentId,
    kind
  }

  // 4. 注册到AppState
  registerTask(taskState, setAppState)

  // 5. 后台化ShellCommand
  shellCommand.background(taskId)

  // 6. 启动停滞监控
  const cancelStallWatchdog = startStallWatchdog(
    taskId, description, kind, toolUseId, agentId
  )

  // 7. 等待结果并处理完成
  void shellCommand.result.then(async result => {
    cancelStallWatchdog()
    await flushAndCleanup(shellCommand)
    
    // 更新状态
    updateTaskState<LocalShellTaskState>(taskId, setAppState, task => {
      if (task.status === 'killed') return task
      return {
        ...task,
        status: result.code === 0 ? 'completed' : 'failed',
        result: { code: result.code, interrupted: result.interrupted },
        shellCommand: null,
        unregisterCleanup: undefined,
        endTime: Date.now()
      }
    })
    
    // 发送通知
    enqueueShellNotification(taskId, description, ...)
    void evictTaskOutput(taskId)
  })

  return { taskId, cleanup: unregisterCleanup }
}
```

### 3.4 任务终止机制

**文件位置:** `tasks/LocalShellTask/killShellTasks.ts`

```typescript
export function killTask(
  taskId: string,
  setAppState: SetAppStateFn
): void {
  updateTaskState(taskId, setAppState, task => {
    if (task.status !== 'running' || !isLocalShellTask(task)) {
      return task
    }

    try {
      task.shellCommand?.kill()    // 终止进程
      task.shellCommand?.cleanup() // 清理资源
    } catch (error) {
      logError(error)
    }

    task.unregisterCleanup?.()     // 取消清理注册
    if (task.cleanupTimeoutId) {
      clearTimeout(task.cleanupTimeoutId)
    }

    return {
      ...task,
      status: 'killed',
      notified: true,
      shellCommand: null,
      endTime: Date.now(),
    }
  })
  
  void evictTaskOutput(taskId)  // 清理磁盘输出
}

// 终止Agent关联的所有Shell任务
export function killShellTasksForAgent(
  agentId: AgentId,
  getAppState: () => AppState,
  setAppState: SetAppStateFn
): void {
  const tasks = getAppState().tasks ?? {}
  
  for (const [taskId, task] of Object.entries(tasks)) {
    if (
      isLocalShellTask(task) &&
      task.agentId === agentId &&
      task.status === 'running'
    ) {
      killTask(taskId, setAppState)
    }
  }
  
  // 清除该Agent的队列消息
  dequeueAllMatching(cmd => cmd.agentId === agentId)
}
```

### 3.5 停滞监控

```typescript
// 检测交互式提示模式
const PROMPT_PATTERNS = [
  /\(y\/n\)/i,
  /\[y\/n\]/i,
  /\(yes\/no\)/i,
  /\b(?:Do you|Would you|Shall I|Are you sure|Ready to)\b.*\? *$/i,
  /Press (any key|Enter)/i,
  /Continue\?/i,
  /Overwrite\?/i
]

export function looksLikePrompt(tail: string): boolean {
  const lastLine = tail.trimEnd().split('\n').pop() ?? ''
  return PROMPT_PATTERNS.some(p => p.test(lastLine))
}

// 停滞监控看门狗
function startStallWatchdog(
  taskId: string,
  description: string,
  kind: BashTaskKind | undefined,
  toolUseId?: string,
  agentId?: AgentId
): () => void {
  if (kind === 'monitor') return () => {}
  
  let lastSize = 0
  let lastGrowth = Date.now()
  let cancelled = false
  
  const timer = setInterval(() => {
    void stat(outputPath).then(s => {
      if (s.size > lastSize) {
        lastSize = s.size
        lastGrowth = Date.now()
        return
      }
      
      // 45秒无增长
      if (Date.now() - lastGrowth < STALL_THRESHOLD_MS) return
      
      // 检查尾部是否像交互提示
      void tailFile(outputPath, STALL_TAIL_BYTES).then(({ content }) => {
        if (!looksLikePrompt(content)) {
          lastGrowth = Date.now()  // 重置，继续监控
          return
        }
        
        // 发送停滞通知
        enqueuePendingNotification({
          value: `任务 "${description}" 等待交互输入...`,
          mode: 'task-notification',
          priority: 'next',
          agentId
        })
      })
    })
  }, STALL_CHECK_INTERVAL_MS)
  
  return () => { cancelled = true; clearInterval(timer) }
}
```

```typescript
// 标准轮询间隔
export const POLL_INTERVAL_MS = 1000

// 终止任务显示时长
export const STOPPED_DISPLAY_MS = 3_000

// 面板宽限期
export const PANEL_GRACE_MS = 30_000

// 更新任务状态
export function updateTaskState<T extends TaskState>(
  taskId: string,
  setAppState: SetAppState,
  updater: (task: T) => T
): void {
  setAppState(prev => {
    const task = prev.tasks?.[taskId] as T | undefined
    if (!task) return prev
    
    const updated = updater(task)
    if (updated === task) return prev  // 无变化则跳过
    
    return {
      ...prev,
      tasks: { ...prev.tasks, [taskId]: updated }
    }
  })
}

// 注册新任务
export function registerTask(
  task: TaskState,
  setAppState: SetAppState
): void {
  let isReplacement = false
  
  setAppState(prev => {
    const existing = prev.tasks[task.id]
    isReplacement = existing !== undefined
    
    // 替换时保留UI状态
    const merged = existing && 'retain' in existing
      ? { ...task, retain: existing.retain, startTime: existing.startTime }
      : task
    
    return { ...prev, tasks: { ...prev.tasks, [task.id]: merged } }
  })
  
  // 发送SDK事件
  if (!isReplacement) {
    enqueueSdkEvent({
      type: 'system',
      subtype: 'task_started',
      task_id: task.id,
      tool_use_id: task.toolUseId,
      description: task.description,
      task_type: task.type,
    })
  }
}

// 清理终止任务
export function evictTerminalTask(
  taskId: string,
  setAppState: SetAppState
): void {
  setAppState(prev => {
    const task = prev.tasks?.[taskId]
    if (!task) return prev
    if (!isTerminalTaskStatus(task.status)) return prev
    if (!task.notified) return prev
    
    // 检查宽限期
    if ('retain' in task && (task.evictAfter ?? Infinity) > Date.now()) {
      return prev
    }
    
    const { [taskId]: _, ...remainingTasks } = prev.tasks
    return { ...prev, tasks: remainingTasks }
  })
}
```

### 4.2 任务轮询机制

```typescript
// 生成任务附件和更新
export async function generateTaskAttachments(state: AppState): Promise<{
  attachments: TaskAttachment[]
  updatedTaskOffsets: Record<string, number>
  evictedTaskIds: string[]
}> {
  const attachments: TaskAttachment[] = []
  const updatedTaskOffsets: Record<string, number> = {}
  const evictedTaskIds: string[] = []

  for (const taskState of Object.values(state.tasks)) {
    // 已通知的终止任务 -> 清理
    if (taskState.notified) {
      switch (taskState.status) {
        case 'completed':
        case 'failed':
        case 'killed':
          evictedTaskIds.push(taskState.id)
          continue
        case 'pending':
          continue  // 保持
        case 'running':
          break
      }
    }

    // 运行中的任务 -> 获取输出增量
    if (taskState.status === 'running') {
      const delta = await getTaskOutputDelta(
        taskState.id,
        taskState.outputOffset
      )
      if (delta.content) {
        updatedTaskOffsets[taskState.id] = delta.newOffset
      }
    }
  }

  return { attachments, updatedTaskOffsets, evictedTaskIds }
}

// 主轮询循环
export async function pollTasks(
  getAppState: () => AppState,
  setAppState: SetAppState
): Promise<void> {
  const state = getAppState()
  const { attachments, updatedTaskOffsets, evictedTaskIds } =
    await generateTaskAttachments(state)

  applyTaskOffsetsAndEvictions(setAppState, updatedTaskOffsets, evictedTaskIds)

  for (const attachment of attachments) {
    enqueueTaskNotification(attachment)
  }
}
```

**文件位置:** `utils/task/diskOutput.ts`

```typescript
// 最大输出限制（5GB）
export const MAX_TASK_OUTPUT_BYTES = 5 * 1024 * 1024 * 1024

// 获取任务输出目录
export function getTaskOutputDir(): string {
  // 使用项目临时目录 + sessionId
  return join(getProjectTempDir(), getSessionId(), 'tasks')
}

// 获取输出文件路径
export function getTaskOutputPath(taskId: string): string {
  return join(getTaskOutputDir(), `${taskId}.output`)
}

// 磁盘输出写入器
export class DiskTaskOutput {
  #path: string
  #fileHandle: FileHandle | null = null
  #queue: string[] = []
  #bytesWritten = 0
  #capped = false

  constructor(taskId: string) {
    this.#path = getTaskOutputPath(taskId)
  }

  append(content: string): void {
    if (this.#capped) return
    
    this.#bytesWritten += content.length
    if (this.#bytesWritten > MAX_TASK_OUTPUT_BYTES) {
      this.#capped = true
      this.#queue.push('\n[output truncated: exceeded 5GB disk cap]\n')
    } else {
      this.#queue.push(content)
    }
    
    // 启动后台写入循环
    if (!this.#flushPromise) {
      void this.#drain()
    }
  }

  async #drain(): Promise<void> {
    while (this.#queue.length) {
      if (!this.#fileHandle) {
        await ensureOutputDir()
        this.#fileHandle = await open(this.#path, 'a')  // O_NOFOLLOW安全
      }
      await this.#writeAllChunks()
    }
    await this.#fileHandle?.close()
  }

  #writeAllChunks(): Promise<void> {
    // 关键：不要await这里！会导致内存膨胀
    return this.#fileHandle!.appendFile(this.#queueToBuffers())
  }
}

// 获取输出增量
export async function getTaskOutputDelta(
  taskId: string,
  fromOffset: number,
  maxBytes: number = DEFAULT_MAX_READ_BYTES
): Promise<{ content: string; newOffset: number }> {
  const result = await readFileRange(
    getTaskOutputPath(taskId),
    fromOffset,
    maxBytes
  )
  return {
    content: result.content,
    newOffset: fromOffset + result.bytesRead
  }
}
```

```typescript
        <Text>
          {truncate(task.command, activityLimit, true)}
          <ShellProgress shell={task} />
        </Text>
      )
    
    case 'remote_agent':
      const running = task.status === 'running'
      return (
        <Text>
          <Text dimColor>{running ? DIAMOND_OPEN : DIAMOND_FILLED}</Text>
          {truncate(task.title, activityLimit, true)}
          <RemoteSessionProgress session={task} />
        </Text>
      )
    
    case 'dream':
      return (
        <Text>
          {task.description}
          <Text dimColor>· {task.phase} · {detail}</Text>
          <TaskStatusText status={task.status} />
        </Text>
      )
  }
}
```

```typescript
// 1. 定义基础类型
type TaskStatus = 'pending' | 'running' | 'completed' | 'failed' | 'killed'
type TaskType = 'shell' | 'agent'

interface TaskState {
  id: string
  type: TaskType
  status: TaskStatus
  description: string
  startTime: number
  endTime?: number
}

// 2. 创建任务存储
class TaskManager {
  private tasks: Map<string, TaskState> = new Map()
  
  register(task: TaskState) {
    this.tasks.set(task.id, task)
    this.emit('registered', task)
  }
  
  update(taskId: string, updater: (task: TaskState) => TaskState) {
    const task = this.tasks.get(taskId)
    if (task) {
      const updated = updater(task)
      this.tasks.set(taskId, updated)
      this.emit('updated', updated)
    }
  }
  
  kill(taskId: string) {
    const task = this.tasks.get(taskId)
    if (task && task.status === 'running') {
      this.update(taskId, t => ({ ...t, status: 'killed', endTime: Date.now() }))
      // 终止实际进程...
    }
  }
}

// 3. Shell任务实现
async function spawnShell(command: string, manager: TaskManager): Promise<string> {
  const taskId = generateId('shell')
  
  manager.register({
    id: taskId,
    type: 'shell',
    status: 'running',
    description: command,
    startTime: Date.now()
  })
  
  const process = spawn(command)
  
  process.on('exit', (code) => {
    manager.update(taskId, t => ({
      ...t,
      status: code === 0 ? 'completed' : 'failed',
      endTime: Date.now()
    }))
  })
  
  return taskId
}
```
