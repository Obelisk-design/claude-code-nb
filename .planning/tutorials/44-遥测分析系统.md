# 44-遥测分析系统详解

## 一、遥测系统概述

Claude Code CLI的遥测分析系统是一个**多后端、分层架构**的事件收集与分析框架，用于收集产品使用数据、监控系统健康状态、支持A/B测试实验。

### 1.2 核心设计理念

1. **无依赖设计**：`index.ts`零依赖，避免循环导入，事件先入队列
2. **延迟绑定**：Sink在应用启动时才绑定，队列事件异步释放
3. **隐私优先**：多层隐私保护，PII数据标记与过滤
4. **降级容错**：磁盘缓存失败事件，二次方退避重试

### 1.3 三大后端系统

| 后端 | 文件 | 用途 | 数据存储 |
|------|------|------|----------|
| Datadog | `datadog.ts` | 运维监控、错误追踪 | Datadog云平台 |
| 1P Event Logging | `firstPartyEventLogger.ts` | 产品分析、用户行为 | BigQuery |
| GrowthBook | `growthbook.ts` | A/B测试、特性开关 | 内存+磁盘缓存 |

**关键代码分析**：

```typescript
// 事件队列 - Sink绑定前的事件缓冲
const eventQueue: QueuedEvent[] = []

// Sink接口定义
export type AnalyticsSink = {
  logEvent: (eventName: string, metadata: LogEventMetadata) => void
  logEventAsync: (eventName: string, metadata: LogEventMetadata) => Promise<void>
}

// Sink实例 - 启动时绑定
let sink: AnalyticsSink | null = null

/**
 * 绑定Sink（应用启动时调用）
 * 队列事件通过queueMicrotask异步释放，避免阻塞启动
 */
export function attachAnalyticsSink(newSink: AnalyticsSink): void {
  if (sink !== null) return  // 幂等：只绑定一次
  
  sink = newSink
  
  if (eventQueue.length > 0) {
    const queuedEvents = [...eventQueue]
    eventQueue.length = 0
    
    queueMicrotask(() => {
      for (const event of queuedEvents) {
        if (event.async) {
          void sink!.logEventAsync(event.eventName, event.metadata)
        } else {
          sink!.logEvent(event.eventName, event.metadata)
        }
      }
    })
  }
}

/**
 * 同步事件日志 - 最常用的API
 * Sink未绑定时入队列，否则直接发送
 */
export function logEvent(eventName: string, metadata: LogEventMetadata): void {
  if (sink === null) {
    eventQueue.push({ eventName, metadata, async: false })
    return
  }
  sink.logEvent(eventName, metadata)
}
```

**PII标记类型**：

```typescript
/**
 * 标记类型：确保字符串不含敏感数据
 * 使用：myString as AnalyticsMetadata_I_VERIFIED_THIS_IS_NOT_CODE_OR_FILEPATHS
 */
export type AnalyticsMetadata_I_VERIFIED_THIS_IS_NOT_CODE_OR_FILEPATHS = never

/**
 * 标记类型：PII敏感数据，仅用于特权BQ列
 * _PROTO_*键在Datadog发送前会被stripProtoFields()移除
 */
export type AnalyticsMetadata_I_VERIFIED_THIS_IS_PII_TAGGED = never

/**
 * 移除_PROTO_*键（PII敏感数据过滤）
 */
export function stripProtoFields<V>(metadata: Record<string, V>): Record<string, V> {
  let result: Record<string, V> | undefined
  for (const key in metadata) {
    if (key.startsWith('_PROTO_')) {
      if (result === undefined) result = { ...metadata }
      delete result[key]
    }
  }
  return result ?? metadata
}
```

## 三、GrowthBook集成

### 3.1 GrowthBook概述

GrowthBook是一个**特性开关和A/B测试平台**，Claude Code通过远程评估（remoteEval）获取特性配置。

**核心功能**：
- 特性开关（Feature Gates）：控制功能启用/禁用
- 动态配置（Dynamic Configs）：返回配置对象
- A/B实验（Experiments）：随机分配用户到实验组

### 3.2 初始化流程

```typescript
/**
 * 初始化GrowthBook客户端
 * 使用memoize确保只初始化一次
 */
export const initializeGrowthBook = memoize(
  async (): Promise<GrowthBook | null> => {
    if (!isGrowthBookEnabled()) return null
    
    const attributes = getUserAttributes()
    const clientKey = getGrowthBookClientKey()
    
    // 创建客户端
    const gbClient = new GrowthBook({
      apiHost: 'https://api.anthropic.com/',
      clientKey,
      attributes,
      remoteEval: true,  // 远程评估模式
      cacheKeyAttributes: ['id', 'organizationUUID'],  // 缓存键
      ...(hasAuth ? { apiHostRequestHeaders: authHeaders.headers } : {}),
    })
    
    client = gbClient
    
    // 初始化（超时5秒）
    const result = await gbClient.init({ timeout: 5000 })
    
    // 处理远程评估响应
    const hadFeatures = await processRemoteEvalPayload(gbClient)
    
    if (hadFeatures) {
      syncRemoteEvalToDisk()  // 同步到磁盘缓存
      refreshed.emit()        // 通知订阅者
    }
    
    return gbClient
  }
)
```

### 3.5 磁盘缓存机制

**目的**：网络失败时降级到磁盘缓存

```typescript
/**
 * 处理远程评估响应
 * 将API返回的features存入内存缓存
 */
async function processRemoteEvalPayload(gbClient: GrowthBook): Promise<boolean> {
  const payload = gbClient.getPayload()
  
  if (!payload?.features || Object.keys(payload.features).length === 0) {
    return false  // 空响应，保持现有缓存
  }
  
  // 清除旧数据
  experimentDataByFeature.clear()
  remoteEvalFeatureValues.clear()
  
  // 处理每个特性
  for (const [key, feature] of Object.entries(payload.features)) {
    const v = 'value' in feature ? feature.value : feature.defaultValue
    if (v !== undefined) {
      remoteEvalFeatureValues.set(key, v)
    }
  }
  
  return true
}

/**
 * 同步到磁盘缓存
 * 存储在 ~/.claude.json 的 cachedGrowthBookFeatures 字段
 */
function syncRemoteEvalToDisk(): void {
  const fresh = Object.fromEntries(remoteEvalFeatureValues)
  saveGlobalConfig(current => ({
    ...current,
    cachedGrowthBookFeatures: fresh,
  }))
}
```

```typescript
let logBatch: DatadogLog[] = []
let flushTimer: NodeJS.Timeout | null = null

/**
 * 刷新日志到Datadog
 */
async function flushLogs(): Promise<void> {
  if (logBatch.length === 0) return
  
  const logsToSend = logBatch
  logBatch = []
  
  try {
    await axios.post(DATADOG_LOGS_ENDPOINT, logsToSend, {
      headers: {
        'Content-Type': 'application/json',
        'DD-API-KEY': DATADOG_CLIENT_TOKEN,
      },
      timeout: NETWORK_TIMEOUT_MS,
    })
  } catch (error) {
    logError(error)
  }
}

/**
 * 调度刷新
 */
function scheduleFlush(): void {
  if (flushTimer) return
  
  flushTimer = setTimeout(() => {
    flushTimer = null
    void flushLogs()
  }, getFlushIntervalMs()).unref()  // unref防止阻塞进程退出
}
```

### 4.5 事件追踪实现

```typescript
export async function trackDatadogEvent(
  eventName: string,
  properties: { [key: string]: boolean | number | undefined }
): Promise<void> {
  // 1. 仅生产环境
  if (process.env.NODE_ENV !== 'production') return
  
  // 2. 仅第一方提供商（非Bedrock/Vertex）
  if (getAPIProvider() !== 'firstParty') return
  
  // 3. 检查初始化和事件白名单
  if (!initialized || !DATADOG_ALLOWED_EVENTS.has(eventName)) return
  
  // 4. 获取元数据
  const metadata = await getEventMetadata({ model: properties.model })
  
  // 5. 构建日志对象
  const log: DatadogLog = {
    ddsource: 'nodejs',
    ddtags: `event:${eventName},model:${metadata.model}`,
    message: eventName,
    service: 'claude-code',
    hostname: 'claude-code',
  }
  
  // 6. 加入批次
  logBatch.push(log)
  
  // 7. 批次满则立即刷新，否则调度刷新
  if (logBatch.length >= MAX_BATCH_SIZE) {
    void flushLogs()
  } else {
    scheduleFlush()
  }
}
```

## 五、1P Event Logging详解

### 5.1 概述

1P（First-Party）事件日志系统是**Anthropic内部产品分析平台**，收集用户行为数据到BigQuery。

### 5.2 OpenTelemetry架构

**基于OpenTelemetry Logs SDK构建**：

```typescript
import { LoggerProvider, BatchLogRecordProcessor } from '@opentelemetry/sdk-logs'

let firstPartyEventLogger: Logger | null = null
let firstPartyEventLoggerProvider: LoggerProvider | null = null

/**
 * 初始化1P事件日志
 */
export function initialize1PEventLogging(): void {
  if (!is1PEventLoggingEnabled()) return

  const batchConfig = getBatchConfig()
  
  // 2. 创建资源（服务名、版本、平台）

  const eventLoggingExporter = new FirstPartyEventLoggingExporter({

    isKilled: () => isSinkKilled('firstParty'),

  // 4. 创建LoggerProvider
  firstPartyEventLoggerProvider = new LoggerProvider({
    resource,
    processors: [
      new BatchLogRecordProcessor(eventLoggingExporter, {
        scheduledDelayMillis: 10000,  // 10秒刷新
        maxExportBatchSize: 200,
        maxQueueSize: 8192,
      }),
    ],
  })
  
  // 5. 获取Logger
  firstPartyEventLogger = firstPartyEventLoggerProvider.getLogger(
    'com.anthropic.claude_code.events',
    MACRO.VERSION,
  )
}
```

### 5.3 事件采样机制

```typescript
const EVENT_SAMPLING_CONFIG_NAME = 'tengu_event_sampling_config'

export type EventSamplingConfig = {
  [eventName: string]: { sample_rate: number }
}

/**
 * 采样配置示例（GrowthBook动态配置）：
 * {
 *   "tengu_api_query": { "sample_rate": 0.1 },  // 10%采样
 *   "tengu_tool_use_success": { "sample_rate": 0.5 }  // 50%采样
 * }
 */

export function shouldSampleEvent(eventName: string): number | null {
  const config = getEventSamplingConfig()
  const eventConfig = config[eventName]
  
  if (!eventConfig) return null  // 未配置则100%记录
  
  const sampleRate = eventConfig.sample_rate
  
  if (sampleRate >= 1) return null   // 100%记录
  if (sampleRate <= 0) return 0      // 丢弃
  
  // 随机采样
  return Math.random() < sampleRate ? sampleRate : 0
}
```

**核心职责**：
- 将日志记录转换为Proto格式
- 通过HTTP发送到`/api/event_logging/batch`
- 失败事件磁盘持久化+二次方退避重试

```typescript
export class FirstPartyEventLoggingExporter implements LogRecordExporter {
  private readonly endpoint: string
  private readonly maxAttempts: number = 8
  private attempts = 0
  
  constructor(options = {}) {
    // 默认prod端点
    this.endpoint = `${baseUrl}/api/event_logging/batch`
  }
  
  /**
   * 导出日志批次
   */
  async export(
    logs: ReadableLogRecord[],
    resultCallback: (result: ExportResult) => void
  ): Promise<void> {
    // 1. 过滤事件日志
    const eventLogs = logs.filter(
      log => log.instrumentationScope?.name === 'com.anthropic.claude_code.events'
    )
    
    // 2. 转换为Proto事件
    const events = this.transformLogsToEvents(eventLogs)
    
    // 3. 发送批次
    const failedEvents = await this.sendEventsInBatches(events)
    
    // 4. 失败事件入队磁盘
    if (failedEvents.length > 0) {
      await this.queueFailedEvents(failedEvents)
      this.scheduleBackoffRetry()
    }
  }
  
  /**
   * 转换日志到事件
   */
  private transformLogsToEvents(logs: ReadableLogRecord[]): FirstPartyEventLoggingPayload {
    const events: FirstPartyEventLoggingEvent[] = []
    
    for (const log of logs) {
      const attributes = log.attributes || {}
      
      // 提取事件名
      const eventName = attributes.event_name as string
      
      // 转换为Proto格式
      events.push({
        event_type: 'ClaudeCodeInternalEvent',
        event_data: ClaudeCodeInternalEvent.toJSON({
          event_id: attributes.event_id,
          event_name: eventName,
          client_timestamp: this.hrTimeToDate(log.hrTime),
          session_id: getSessionId(),
          ...formatted.core,
          env: formatted.env,
          additional_metadata: Buffer.from(jsonStringify(additionalMetadata)).toString('base64'),
        }),
      })
    }
    
    return { events }
  }
  
  /**
   * 二次方退避重试
   */
  private scheduleBackoffRetry(): void {
    // delay = base * attempts²
    const delay = Math.min(
      this.baseBackoffDelayMs * this.attempts * this.attempts,
      this.maxBackoffDelayMs  // 最大30秒
    )
    
    this.cancelBackoff = this.schedule(async () => {
      await this.retryFailedEvents()
    }, delay)
  }
}
```

```typescript
let prevCpuUsage: NodeJS.CpuUsage | null = null
let prevWallTimeMs: number | null = null

function buildProcessMetrics(): ProcessMetrics | undefined {
  const mem = process.memoryUsage()
  const cpu = process.cpuUsage()
  const now = Date.now()
  
  // 计算CPU百分比增量
  let cpuPercent: number | undefined
  if (prevCpuUsage && prevWallTimeMs) {
    const wallDeltaMs = now - prevWallTimeMs
    const userDeltaUs = cpu.user - prevCpuUsage.user
    const systemDeltaUs = cpu.system - prevCpuUsage.system
    cpuPercent = ((userDeltaUs + systemDeltaUs) / (wallDeltaMs * 1000)) * 100
  }
  
  prevCpuUsage = cpu
  prevWallTimeMs = now
  
  return { uptime: process.uptime(), ...mem, cpuUsage: cpu, cpuPercent }
}
```

### 6.7 文件扩展名提取

```typescript
const MAX_FILE_EXTENSION_LENGTH = 10

/**
 * 提取文件扩展名（用于分析文件类型分布）
 */
export function getFileExtensionForAnalytics(
  filePath: string
): AnalyticsMetadata_I_VERIFIED_THIS_IS_NOT_CODE_OR_FILEPATHS | undefined
{
  const ext = extname(filePath).toLowerCase()
  if (!ext || ext === '.') return undefined
  
  const extension = ext.slice(1)  // 移除前导点
  if (extension.length > MAX_FILE_EXTENSION_LENGTH) {
    return 'other'  // 过长扩展名视为敏感
  }
  
  return extension
}

/**
 * 从Bash命令提取文件扩展名
 */
export function getFileExtensionsFromBashCommand(
  command: string
): AnalyticsMetadata_I_VERIFIED_THIS_IS_NOT_CODE_OR_FILEPATHS | undefined
{
  // 分割复合命令：&&, ||, ;, |
  for (const subcmd of command.split(/\s*(?:&&|\|\||[;|])\s*/)) {
    const tokens = subcmd.split(/\s+/)
    const baseCmd = tokens[0]?.split('/').pop()
    
    // 仅处理文件操作命令
    if (!FILE_COMMANDS.has(baseCmd)) continue
    
    // 提取非flag参数的扩展名
    for (const arg of tokens.slice(1)) {
      if (arg.startsWith('-')) continue
      const ext = getFileExtensionForAnalytics(arg)
      if (ext) return ext
    }
  }
}
```

```typescript
// services/analytics/index.ts

type LogEventMetadata = { [key: string]: boolean | number | undefined }

type AnalyticsSink = {
  logEvent: (eventName: string, metadata: LogEventMetadata) => void
}

const eventQueue: { eventName: string, metadata: LogEventMetadata }[] = []
let sink: AnalyticsSink | null = null

export function attachAnalyticsSink(newSink: AnalyticsSink): void {
  if (sink !== null) return
  sink = newSink
  
  // 异步释放队列事件
  if (eventQueue.length > 0) {
    const queued = [...eventQueue]
    eventQueue.length = 0
    queueMicrotask(() => {
      for (const e of queued) sink!.logEvent(e.eventName, e.metadata)
    })
  }
}

export function logEvent(eventName: string, metadata: LogEventMetadata): void {
  if (sink === null) {
    eventQueue.push({ eventName, metadata })
    return
  }
  sink.logEvent(eventName, metadata)
}
```

```typescript
import axios from 'axios'

const DATADOG_ENDPOINT = 'https://http-intake.logs.datadoghq.com/api/v2/logs'
const CLIENT_TOKEN = 'your-token'

let logBatch: any[] = []
let flushTimer: NodeJS.Timeout | null = null

async function flushLogs(): Promise<void> {
  if (logBatch.length === 0) return
  const logs = logBatch
  logBatch = []
  
  await axios.post(DATADOG_ENDPOINT, logs, {
    headers: { 'DD-API-KEY': CLIENT_TOKEN },
    timeout: 5000,
  })
}

function scheduleFlush(): void {
  if (flushTimer) return
  flushTimer = setTimeout(() => {
    flushTimer = null
    void flushLogs()
  }, 15000).unref()
}

export async function trackDatadogEvent(
  eventName: string,
  metadata: Record<string, any>
): Promise<void> {
  const log = {
    ddsource: 'nodejs',
    message: eventName,
    service: 'my-cli',
    ...metadata,
  }
  
  logBatch.push(log)
  
  if (logBatch.length >= 100) {
    void flushLogs()
  } else {
    scheduleFlush()
  }
}
```
