**分析日期:** 2026-04-02

## 1. 剪贴板概述

Claude Code CLI 的剪贴板系统是一个精心设计的跨平台解决方案，它解决了终端环境下剪贴板操作的独特挑战。与 GUI 应用不同，终端应用无法直接访问系统剪贴板，必须通过特殊的协议和工具来实现。

## 2. 跨平台实现

### 2.1 OSC 52 协议详解

OSC 52 是终端的标准剪贴板协议，允许终端应用通过转义序列写入系统剪贴板。

**协议格式：**
```
ESC ] 52 ; <clipboard> ; <base64-data> BEL
```

- `<clipboard>`: 剪贴板选择器 (`c` = 系统剪贴板, `p` = 主选择区 X11)
- `<base64-data>`: Base64 编码的剪贴板内容
- `BEL` (0x07) 或 `ST` (ESC \): 序列终止符

**核心实现 (`ink/termio/osc.ts`):**

```typescript
// 第 138-158 行
export async function setClipboard(text: string): Promise<string> {
  const b64 = Buffer.from(text, 'utf8').toString('base64')
  const raw = osc(OSC.CLIPBOARD, 'c', b64)  // 生成 OSC 52 序列

  // 原生安全网 — 在 OSC 52 之前启动，避免竞态条件
  // SSH 环境下不执行（会写入远程机器剪贴板）
  if (!process.env['SSH_CONNECTION']) copyNative(text)

  const tmuxBufferLoaded = await tmuxLoadBuffer(text)

  // tmux 环境下返回 DCS 包装的 OSC 52
  if (tmuxBufferLoaded) return tmuxPassthrough(`${ESC}]52;c;${b64}${BEL}`)
  return raw  // 直接返回原始 OSC 52
}
```

### 2.2 剪贴板路径决策

系统根据环境自动选择最佳剪贴板路径：

```typescript
// 第 64-70 行
export function getClipboardPath(): ClipboardPath {
  const nativeAvailable =
    process.platform === 'darwin' && !process.env['SSH_CONNECTION']
  if (nativeAvailable) return 'native'       // macOS 本地: pbcopy
  if (process.env['TMUX']) return 'tmux-buffer'  // tmux 环境

}
```

**三种路径详解：**

| 路径 | 条件 | 实现 | 特点 |
|-----|------|-----|------|
| `native` | macOS + 无SSH | `pbcopy` | 高置信度，始终有效 |
| `tmux-buffer` | $TMUX 设置 | `tmux load-buffer -w` | 支持断开/重连，prefix+] 粘贴 |
| `osc52` | 其他情况 | OSC 52 序列 | 最佳努力，依赖终端支持 |

### 2.3 tmux 集成

tmux 会拦截 OSC 52 序列，需要特殊处理：

```typescript
// 第 90-102 行
export async function tmuxLoadBuffer(text: string): Promise<boolean> {
  if (!process.env['TMUX']) return false
  
  // -w 标志让 tmux 将内容传播到外层终端剪贴板
  // iTerm2 例外：tmux 的 OSC 52 会导致崩溃
  const args =
    process.env['LC_TERMINAL'] === 'iTerm2'
      ? ['load-buffer', '-']
      : ['load-buffer', '-w', '-']
  
  const { code } = await execFileNoThrow('tmux', args, {
    input: text,
    useCwd: false,
    timeout: 2000,
  })
  return code === 0
}
```

```typescript
function copyNative(text: string): void {
  const opts = { input: text, useCwd: false, timeout: 2000 }
  
  switch (process.platform) {
    case 'darwin':
      void execFileNoThrow('pbcopy', [], opts)  // macOS
      return
    
    case 'linux': {
      // 探测顺序: wl-copy (Wayland) → xclip (X11) → xsel (X11 fallback)

        void execFileNoThrow('wl-copy', [], opts)
        return
      }
      if (linuxCopy === 'xclip') {
        void execFileNoThrow('xclip', ['-selection', 'clipboard'], opts)
        return
      }
      // ... 探测逻辑
    }
    
    case 'win32':
      void execFileNoThrow('clip', [], opts)  // Windows clip.exe
      return
  }
}
```

```typescript
// 第 50-61 行
export function collectRecentAssistantTexts(messages: Message[]): string[] {
  const texts: string[] = []
  // 从最新到最旧遍历，最多 MAX_LOOKBACK (20) 条
  for (let i = messages.length - 1; i >= 0 && texts.length < MAX_LOOKBACK; i--) {
    const msg = messages[i]
    // 跳过非助手消息和 API 错误消息
    if (msg?.type !== 'assistant' || msg.isApiErrorMessage) continue
    const content = (msg as AssistantMessage).message.content
    if (!Array.isArray(content)) continue
    const text = extractTextContent(content, '\n\n')
    if (text) texts.push(text)
  }
  return texts
}
```

```typescript
// 第 81-94 行
async function copyOrWriteToFile(text: string, filename: string): Promise<string> {
  const raw = await setClipboard(text)   // OSC 52 写入
  if (raw) process.stdout.write(raw)     // 输出序列到终端
  
  const lineCount = countCharInString(text, '\n') + 1
  const charCount = text.length
  
  // 同时写入临时文件作为备份（OSC 52 可能不被支持）
  try {
    const filePath = await writeToFile(text, filename)
    return `Copied to clipboard (${charCount} characters, ${lineCount} lines)\nAlso written to ${filePath}`
  } catch {
    return `Copied to clipboard (${charCount} characters, ${lineCount} lines)`
  }
}
```

### 3.4 命令调用入口

```typescript
// 第 334-370 行
export const call: LocalJSXCommandCall = async (onDone, context, args) => {
  const texts = collectRecentAssistantTexts(context.messages)
  
  if (texts.length === 0) {
    onDone('No assistant message to copy')
    return null
  }

  // /copy N 语法：N=1 最新, N=2 第二新, ...
  let age = 0
  const arg = args?.trim()
  if (arg) {
    const n = Number(arg)
    if (!Number.isInteger(n) || n < 1) {
      onDone(`Usage: /copy [N] where N is 1 (latest), 2, 3, … Got: ${arg}`)
      return null
    }
    if (n > texts.length) {
      onDone(`Only ${texts.length} assistant messages available to copy`)
      return null
    }
    age = n - 1
  }
  
  const text = texts[age]!
  const codeBlocks = extractCodeBlocks(text)
  const config = getGlobalConfig()
  
  // 无代码块或配置为始终复制完整回复 → 直接复制
  if (codeBlocks.length === 0 || config.copyFullResponse) {
    const result = await copyOrWriteToFile(text, RESPONSE_FILENAME)
    onDone(result)
    return null
  }
  
  // 显示选择器
  return <CopyPicker fullText={text} codeBlocks={codeBlocks} messageAge={age} onDone={onDone} />
}
```

```typescript
// 第 293-317 行
function stripBackslashEscapes(path: string): string {
  const platform = process.platform
  
  // Windows 不处理反斜杠（它们是路径的一部分）
  if (platform === 'win32') return path
  
  // macOS/Linux: 处理 shell 转义
  // 双反斜杠 \\ 表示真实的反斜杠
  // 单反斜杠 + 特殊字符是 shell 转义
  const salt = randomBytes(8).toString('hex')
  const placeholder = `__DOUBLE_BACKSLASH_${salt}__`
  const withPlaceholder = path.replace(/\\\\/g, placeholder)
  const withoutEscapes = withPlaceholder.replace(/\\(.)/g, '$1')
  return withoutEscapes.replace(new RegExp(placeholder, 'g'), '\\')
}
```

### 4.3 usePasteHandler 钩子

处理粘贴事件的核心钩子：

```typescript
// hooks/usePasteHandler.ts
export function usePasteHandler({
  onPaste,
  onInput,
  onImagePaste,
}): { wrappedOnInput, pasteState, isPasting } {
  
  // 粘贴阈值（超过此长度视为粘贴）
  const PASTE_THRESHOLD = 800
  
  // macOS 原生剪贴板图片检测
  const checkClipboardForImage = useDebounceCallback(() => {
    void getImageFromClipboard()
      .then(imageData => {
        if (imageData && isMountedRef.current) {
          onImagePaste(imageData.base64, imageData.mediaType, ...)
        }
      })
      .finally(() => setIsPasting(false))
  }, 50)  // 50ms debounce
  
  // 输入处理
  const wrappedOnInput = (input: string, key: Key, event: InputEvent): void => {
    const isFromPaste = event.keypress.isPasted  // Bracketed paste 标记
    
    // 空粘贴 → 检查剪贴板图片（macOS）
    if (isFromPaste && input.length === 0 && isMacOS && onImagePaste) {
      checkClipboardForImage()
      return
    }
    
    // 检测图片文件路径
    const hasImageFilePath = input
      .split(/ (?=\/|[A-Za-z]:\\)/)
      .flatMap(part => part.split('\n'))
      .some(line => isImageFilePath(line.trim()))
    
    // 大输入、粘贴标记、或图片路径 → 紘贴处理
    const shouldHandleAsPaste =
      input.length > PASTE_THRESHOLD ||
      pastePendingRef.current ||
      hasImageFilePath ||
      isFromPaste
    
    if (shouldHandleAsPaste) {
      // 收集 chunks 并设置超时
      setPasteState(({ chunks, timeoutId }) => ({
        chunks: [...chunks, input],
        timeoutId: resetPasteTimeout(timeoutId),
      }))
      return
    }
    
    // 正常输入
    onInput(input, key)
  }
}
```

```typescript
export function useClipboardImageHint(
  isFocused: boolean,
  enabled: boolean,  // onImagePaste 是否定义
): void {
  const FOCUS_CHECK_DEBOUNCE_MS = 1000   // 焦点检查防抖
  const HINT_COOLDOWN_MS = 30000         // 提示冷却时间
  
  useEffect(() => {
    // 仅在重新获得焦点时触发
    const wasFocused = lastFocusedRef.current
    lastFocusedRef.current = isFocused
    
    if (!enabled || !isFocused || wasFocused) return
    
    // 防抖检查
    checkTimeoutRef.current = setTimeout(async () => {
      // 冷却检查
      if (now - lastHintTimeRef.current < HINT_COOLDOWN_MS) return
      
      // 检查剪贴板图片
      if (await hasImageInClipboard()) {
        lastHintTimeRef.current = now
        addNotification({
          key: NOTIFICATION_KEY,
          text: `Image in clipboard · ${getShortcutDisplay('chat:imagePaste', 'Chat', 'ctrl+v')} to paste`,
          priority: 'immediate',
          timeoutMs: 8000,
        })
      }
    }, FOCUS_CHECK_DEBOUNCE_MS)
  }, [isFocused, enabled, addNotification])
}
```

### 5.3 图片粘贴

**支持的粘贴方式：**
1. **剪贴板图片粘贴** - Ctrl+V / Cmd+V
   - macOS: 直接从剪贴板读取 PNG
   - Linux: 通过 xclip/wl-paste 读取
   - Windows: 通过 PowerShell 读取

2. **拖拽图片文件** - 终端支持时

```typescript
async function setClipboardWithFallback(text: string): Promise<boolean> {
  // 1. 首先尝试原生工具（本地环境）
  if (!process.env.SSH_CONNECTION) {
    try {
      if (process.platform === 'darwin') {
        await execa('pbcopy', { input: text })
        return true
      }
      if (process.platform === 'linux') {
        await execa('xclip', ['-selection', 'clipboard'], { input: text })
        return true
      }
      if (process.platform === 'win32') {
        await execa('clip', { input: text })
        return true
      }
    } catch {
      // 继续尝试 OSC 52
    }
  }
  
  // 2. OSC 52 作为后备
  const b64 = Buffer.from(text).toString('base64')
  const sequence = `\x1b]52;c;${b64}\x07`
  
  // tmux 包装
  if (process.env.TMUX) {
    const escaped = sequence.replace(/\x1b/g, '\x1b\x1b')
    process.stdout.write(`\x1bPtmux;${escaped}\x1b\\`)
  } else {
    process.stdout.write(sequence)
  }
  
  return true
}
```

### 6.4 粘贴事件检测

```typescript
function createPasteDetector(onPaste: (text: string) => void) {
  const PASTE_THRESHOLD = 800
  let pasteChunks: string[] = []
  let pasteTimeout: NodeJS.Timeout | null = null
  
  function handleInput(input: string, isBracketedPaste: boolean) {
    // 大输入或 bracketed paste 标记
    const isPaste = input.length > PASTE_THRESHOLD || isBracketedPaste
    
    if (isPaste) {
      pasteChunks.push(input)
      
      // 重置超时
      if (pasteTimeout) clearTimeout(pasteTimeout)
      
      pasteTimeout = setTimeout(() => {
        const pastedText = pasteChunks.join('')
        pasteChunks = []
        pasteTimeout = null
        onPaste(pastedText)
      }, 100)
      
      return
    }
    
    // 正常输入处理...
  }
  
  return { handleInput }
}
```

### 6.5 选择自动复制实现

```typescript
function createSelectionCopyHandler(
  selection: { subscribe: (cb: () => void) => () => void, getState: () => SelectionState, hasSelection: () => boolean, copySelectionNoClear: () => string },
  config: { copyOnSelect: boolean }
) {
  let copied = false
  
  const unsubscribe = selection.subscribe(() => {
    const state = selection.getState()
    const hasSelection = selection.hasSelection()
    
    // 拖拽中或无选区 → 重置标记
    if (state?.isDragging || !hasSelection) {
      copied = false
      return
    }
    
    // 已复制 → 避免重复
    if (copied) return
    
    // 配置检查
    if (!config.copyOnSelect) return
    
    // 复制文本
    const text = selection.copySelectionNoClear()
    if (!text?.trim()) {
      copied = true
      return
    }
    
    copied = true
    setClipboardWithFallback(text)
  })
  
  return { unsubscribe }
}
```

## 7. 关键技术要点

### 7.1 OSC 52 协议细节

**序列构造：**
```typescript
function osc(...parts: (string | number)[]): string {
  const OSC_PREFIX = '\x1b]'  // ESC ]
  const BEL = '\x07'          // Bell
  const ST = '\x1b\\'         // String Terminator (ESC \)
  const SEP = ';'             // 分隔符
  
  // Kitty 使用 ST 终止符（避免蜂鸣声）
  const terminator = env.terminal === 'kitty' ? ST : BEL
  return `${OSC_PREFIX}${parts.join(SEP)}${terminator}`
}
```
